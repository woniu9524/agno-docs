---
title: 编写自己的工具包
---

许多高级用例都需要编写自定义的工具包。以下是通用流程：

1. 创建一个继承自 `agno.tools.Toolkit` 类的类。
2. 将你的函数添加到该类中。
3. **重要提示：** 在 `Toolkit` 构造函数的 `tools` 参数中包含所有函数。

现在你的工具包已准备好与 Agent 一起使用。例如：

```python shell_toolkit.py
from typing import List

from agno.agent import Agent
from agno.tools import Toolkit
from agno.utils.log import logger

class ShellTools(Toolkit):
    def __init__(self, **kwargs):
        super().__init__(name="shell_tools", tools=[self.run_shell_command], **kwargs)

    def run_shell_command(self, args: List[str], tail: int = 100) -> str:
        """
        运行 shell 命令并返回输出或错误。

        Args:
            args (List[str]): 要作为字符串列表运行的命令。
            tail (int): 要从输出中返回的行数。
        Returns:
            str: 命令的输出。
        """
        import subprocess

        logger.info(f"运行 shell 命令: {args}")
        try:
            logger.info(f"运行 shell 命令: {args}")
            result = subprocess.run(args, capture_output=True, text=True)
            logger.debug(f"结果: {result}")
            logger.debug(f"返回码: {result.returncode}")
            if result.returncode != 0:
                return f"错误: {result.stderr}"
            # 只返回输出的最后 n 行
            return "\n".join(result.stdout.split("\n")[-tail:])
        except Exception as e:
            logger.warning(f"无法运行 shell 命令: {e}")
            return f"错误: {e}"

agent = Agent(tools=[ShellTools()], show_tool_calls=True, markdown=True)
agent.print_response("列出我主目录中的所有文件。")
