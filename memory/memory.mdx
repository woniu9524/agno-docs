title: ユーザーメモリ
---

メモリについて話すとき、一般的に合意されているメモリの理解は、エージェントが対話しているユーザーに関する洞察や事実を保存する能力です。簡単に言えば、ユーザーのペルソナを構築し、その好みについて学び、それを利用してエージェントの応答をパーソナライズすることです。

## エージェントメモリ

Agnoエージェントはネイティブでエージェントメモリ管理をサポートしており、メモリジャーニーの出発点として推奨しています。

エージェントメモリを使用すると、エージェント自体がユーザーの会話からメモリを作成、更新、削除します。

`enable_agentic_memory=True` を設定して、エージェントにユーザーのメモリを管理するツールを提供します。このツールはタスクを `MemoryManager` クラスに渡します。

> `enable_user_memories=True` を設定することもできます。これにより、各ユーザーメッセージの後に常に `MemoryManager` が実行されます。[例については以下を参照してください。](#create-memories-after-each-run)

```python agentic_memory.py
from agno.agent import Agent
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.memory.v2.memory import Memory
from agno.models.openai import OpenAIChat
from agno.storage.sqlite import SqliteStorage
from rich.pretty import pprint

# メモリのユーザーID
user_id = "ava"
# メモリとストレージのデータベースファイル
db_file = "tmp/agent.db"

# memory.v2 を初期化
memory = Memory(
    # メモリ作成には任意のモデルを使用
    model=OpenAIChat(id="gpt-4.1"),
    db=SqliteMemoryDb(table_name="user_memories", db_file=db_file),
)
# ストレージを初期化
storage = SqliteStorage(table_name="agent_sessions", db_file=db_file)

# Agent を初期化
memory_agent = Agent(
    model=OpenAIChat(id="gpt-4.1"),
    # メモリをデータベースに保存
    memory=memory,
    # メモリを更新する能力をエージェントに付与
    enable_agentic_memory=True,
    # または - 各応答後にメモリマネージャーを実行
    enable_user_memories=True,
    # チャット履歴をデータベースに保存
    storage=storage,
    # メッセージにチャット履歴を追加
    add_history_to_messages=True,
    # 履歴実行数
    num_history_runs=3,
    markdown=True,
)

memory.clear()
memory_agent.print_response(
    "私の名前はAvaで、スキーが好きです。",
    user_id=user_id,
    stream=True,
    stream_intermediate_steps=True,
)
print("Avaに関するメモリ:")
pprint(memory.get_user_memories(user_id=user_id))

memory_agent.print_response(
    "サンフランシスコに住んでいます。4時間以内の移動でどこに引っ越すべきですか？",
    user_id=user_id,
    stream=True,
    stream_intermediate_steps=True,
)
print("Avaに関するメモリ:")
pprint(memory.get_user_memories(user_id=user_id))
```

- `add_history_to_messages=True` は、モデルに送信されるメッセージにチャット履歴を追加します。`num_history_runs` は追加する実行数を決定します。
- `read_chat_history=True` は、エージェントにチャット履歴を読み取るためのツールを追加します。これは、`num_history_runs` に含まれるものよりも大きい場合があるためです。

## 各実行後にメモリを作成する

`enable_agentic_memory=True` はエージェントにユーザーメモリを管理するツールを提供しますが、各ユーザーメッセージの後に常に `MemoryManagement` を「トリガー」することもできます。

`enable_user_memories=True` を設定すると、各ユーザーメッセージの後に常にメモリが処理されます。

```python create_memories_after_each_run.py
from agno.agent.agent import Agent
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.memory.v2.memory import Memory
from rich.pretty import pprint

memory_db = SqliteMemoryDb(table_name="memory", db_file="tmp/memory.db")
# モデルを設定する必要はありません。エージェントのモデルに設定されます。
memory = Memory(db=memory_db, delete_memories=True, clear_memories=True)

# この例のためにメモリをリセット
memory.clear()

# メモリのユーザーID
john_doe_id = "john_doe@example.com"
agent = Agent(
    model=OpenAIChat(id="gpt-4o-mini"),
    memory=memory,
    # これにより、各ユーザーメッセージの後にメモリマネージャーがトリガーされます。
    enable_user_memories=True,
)

# メモリが使用されるメッセージをエージェントに送信します。
agent.print_response(
    "私の名前はJohn Doeで、週末は山でハイキングするのが好きです。",
    stream=True,
    user_id=john_doe_id,
)

# メモリが機能しているかを確認するメッセージをエージェントに送信します。
agent.print_response("私の趣味は何ですか？", stream=True, user_id=john_doe_id)

# ユーザーのメモリを印刷します。
memories = memory.get_user_memories(user_id=john_doe_id)
print("John Doeに関するメモリ:")
pprint(memories)

# ユーザーのすべてのメモリを削除するメッセージをエージェントに送信します。
agent.print_response(
    "私のすべての既存のメモリを削除してください。",
    stream=True,
    user_id=john_doe_id,
)
memories = memory.get_user_memories(user_id=john_doe_id)
print("John Doeに関するメモリ:")
pprint(memories)
```

## メモリ管理

Agnoの `Memory` クラスを使用すると、ユーザーメモリのすべての側面を管理できます。まず、エージェント以外の `Memory` の使用例を見てみましょう。以下を行います。

- メモリの追加、更新、削除
- メモリをデータベースに保存
- 会話からメモリを作成
- メモリを検索

```python
from agno.memory.v2.memory import Memory
from agno.memory.v2.db.sqlite import SqliteMemoryDb

# 永続ストレージを持つメモリインスタンスを作成します。
memory_db = SqliteMemoryDb(table_name="memory", db_file="memory.db")
memory = Memory(db=memory_db)
```

### 新しいメモリの追加

```python
from agno.memory.v2.memory import Memory
from agno.memory.v2.schema import UserMemory

memory = Memory()

# ユーザーメモリを手動で作成します。
memory_id = memory.add_user_memory(
    memory=UserMemory(
        memory="ユーザーの名前はJane Doeです",
        topics=["personal", "name"]
    ),
    user_id="jane_doe@example.com"
)
```

### メモリの更新

```python
from agno.memory.v2.memory import Memory
from agno.memory.v2.schema import UserMemory

memory = Memory()

# ユーザーメモリを置き換えます。
memory_id = memory.replace_user_memory(
    # 置き換えるメモリのID
    memory_id=previous_memory_id,
    # 置き換える新しいメモリ
    memory=UserMemory(
        memory="ユーザーの名前はVerna Doeです",
        topics=["personal", "name"]
    ),
    user_id="jane_doe@example.com"
)
```

### メモリの削除

```python
from agno.memory.v2.memory import Memory

memory = Memory()

# ユーザーメモリを削除します。
memory.delete_user_memory(user_id="jane_doe@example.com", memory_id=memory_id)
```

### ユーザー情報からメモリを作成する

```python
from agno.memory.v2 import Memory
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.models.google import Gemini

memory_db = SqliteMemoryDb(table_name="memory", db_file="tmp/memory.db")
memory = Memory(model=Gemini(id="gemini-2.0-flash-exp"), db=memory_db)

john_doe_id = "john_doe@example.com"

memory.create_user_memories(
    message="""
    週末は山でハイキングするのが好きで、
    寝る前にSF小説を読み、
    異文化の新しいレシピを調理し、
    友達とチェスをプレイし、
    可能な限りライブ音楽コンサートに参加します。
    特に風景やストリートシーンを捉える写真が最近の情熱になっています。
    朝は瞑想し、ヨガを実践して心を落ち着かせるようにしています。
    """,
    user_id=john_doe_id,
)


memories = memory.get_user_memories(user_id=john_doe_id)
print("John Doeのメモリ:")
for i, m in enumerate(memories):
    print(f"{i}: {m.memory} - {m.topics}")
```

### 会話からメモリを作成する

```python
from agno.memory.v2 import Memory
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.models.google import Gemini
from agno.models.message import Message

memory_db = SqliteMemoryDb(table_name="memory", db_file="tmp/memory.db")
memory = Memory(model=Gemini(id="gemini-2.0-flash-exp"), db=memory_db)


jane_doe_id = "jane_doe@example.com"
# メッセージ履歴を送信し、メモリを追加します。
memory.create_user_memories(
    messages=[
        Message(role="user", content="私の名前はJane Doeです"),
        Message(role="assistant", content="それは素晴らしいですね！"),
        Message(role="user", content="チェスをするのが好きです"),
        Message(role="assistant", content="それは素晴らしいですね！"),
    ],
    user_id=jane_doe_id,
)

memories = memory.get_user_memories(user_id=jane_doe_id)
print("Jane Doeのメモリ:")
for i, m in enumerate(memories):
    print(f"{i}: {m.memory} - {m.topics}")
```

## メモリ検索

Agnoは、ユーザーメモリを検索および取得するためのいくつかの検索方法を提供します。

### 基本的な検索方法

`last_n`（最新）または `first_n`（古い順）のような時系列の方法を使用してメモリを取得できます。

```python
from agno.memory.v2 import Memory, UserMemory

memory = Memory()

john_doe_id = "john_doe@example.com"

memory.add_user_memory(
    memory=UserMemory(memory="ユーザーは週末に山でハイキングすることを楽しんでいます"),
    user_id=john_doe_id,
)
memory.add_user_memory(
    memory=UserMemory(
        memory="ユーザーは寝る前にSF小説を読むことを楽しんでいます"
    ),
    user_id=john_doe_id,
)

# 最新のメモリを取得します。
memories = memory.search_user_memories(
    user_id=john_doe_id, limit=1, retrieval_method="last_n"
)
print("John Doeのlast_nメモリ:")
for i, m in enumerate(memories):
    print(f"{i}: {m.memory}")

# 最も古いメモリを取得します。
memories = memory.search_user_memories(
    user_id=john_doe_id, limit=1, retrieval_method="first_n"
)
print("John Doeのfirst_nメモリ:")
for i, m in enumerate(memories):
    print(f"{i}: {m.memory}")
```

### エージェントメモリ検索

エージェント検索を使用すると、正確なキーワードマッチングではなく、意味に基づいてメモリを見つけることができます。これは、文脈的に関連性の高い情報を取得する場合に特に役立ちます。

```python
from agno.memory.v2.memory import Memory, UserMemory
from agno.models.google.gemini import Gemini

# エージェント検索用のモデルでメモリを初期化します。
memory = Memory(model=Gemini(id="gemini-2.0-flash-exp"))

john_doe_id = "john_doe@example.com"

memory.add_user_memory(
    memory=UserMemory(memory="ユーザーは週末に山でハイキングすることを楽しんでいます"),
    user_id=john_doe_id,
)
memory.add_user_memory(
    memory=UserMemory(
        memory="ユーザーは寝る前にSF小説を読むことを楽しんでいます"
    ),
    user_id=john_doe_id,
)

# クエリに関連するメモリを検索します。
memories = memory.search_user_memories(
    user_id=john_doe_id,
    query="ユーザーは週末に何をすることを楽しんでいますか？",
    retrieval_method="agentic",
)
print("John Doeが見つけたメモリ:")
for i, m in enumerate(memories):
    print(f"{i}: {m.memory}")
```

エージェント検索を使用すると、モデルはクエリの背後にある意図を理解し、検索クエリと正確なキーワードが含まれていない場合でも、最も関連性の高いメモリを返します。
