使用 [Stripe MCP 服务器](https://github.com/stripe/agent-toolkit/tree/main/modelcontextprotocol) 创建一个可以与 Stripe API 交互的 Agent：

```python
"""💵 Stripe MCP Agent - 管理您的 Stripe 操作

此示例演示如何创建与 Stripe API 通过模型上下文协议 (MCP) 交互的 Agno Agent。该 Agent 可以使用自然语言命令创建和管理 Stripe 对象，例如客户、产品、价格和支付链接。


设置：
2. 安装 Python 依赖项：`pip install agno mcp-sdk`
3. 设置环境变量：export STRIPE_SECRET_KEY=***。

Stripe MCP 文档：https://github.com/stripe/agent-toolkit
"""

import asyncio
import os
from textwrap import dedent

from agno.agent import Agent
from agno.tools.mcp import MCPTools
from agno.utils.log import log_error, log_exception, log_info


async def run_agent(message: str) -> None:
    """
    设置 Stripe MCP 服务器并初始化 Agno Agent
    """
    # 验证 Stripe API 密钥是否可用
    stripe_api_key = os.getenv("STRIPE_SECRET_KEY")
    if not stripe_api_key:
        log_error("未设置 STRIPE_SECRET_KEY 环境变量。")
        return

    enabled_tools = "paymentLinks.create,products.create,prices.create,customers.create,customers.read"

    # 处理不同的操作系统
    npx_command = "npx.cmd" if os.name == "nt" else "npx"

    try:
        # 使用 Stripe 服务器初始化 MCP 工具包
        async with MCPTools(
            command=f"{npx_command} -y @stripe/mcp --tools={enabled_tools} --api-key={stripe_api_key}"
        ) as mcp_toolkit:
            agent = Agent(
                name="StripeAgent",
                instructions=dedent("""\
                    您是一位专门管理 Stripe 操作的 AI 助手。
                    您使用可用工具与 Stripe API 进行交互。

                    - 理解用户创建或列出 Stripe 对象（客户、产品、价格、支付链接）的请求。
                    - 清晰说明您的操作结果，包括已创建对象的 ID 或检索到的列表。
                    - 如果请求含糊不清，请寻求澄清。
                    - 使用 Markdown 格式，特别是链接或代码片段。
                    - 如果请求涉及多个操作（例如，先创建产品，然后创建价格，最后创建链接），请按顺序执行必要步骤。
                """),
                tools=[mcp_toolkit],
                markdown=True,
                show_tool_calls=True,
            )

            # 使用提供的任务运行 Agent
            log_info(f"使用分配的任务运行 Agent：'{message}'")
            await agent.aprint_response(message, stream=True)

    except FileNotFoundError:
        error_msg = f"错误：找不到 '{npx_command}' 命令。请确保 Node.js 和 npm/npx 已安装并在您的系统 PATH 中。"
        log_error(error_msg)
    except Exception as e:
        log_exception(f"Agent 执行期间发生意外错误：{e}")


if __name__ == "__main__":
    task = "创建一个名为 'iPhone' 的新 Stripe 产品。然后为其创建一个 999.99 美元的价格。最后，为该价格创建一个支付链接。"
    asyncio.run(run_agent(task))
