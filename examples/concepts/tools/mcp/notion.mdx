[Notion MCP server](https://github.com/makenotion/notion-mcp-server) を使用して、Notion ページを作成、更新、検索できるエージェントを作成します。

```python
"""
Notion MCP Agent - ドキュメントを管理します

この例では、Agno MCP ツールを使用して Notion ワークスペースとやり取りする方法を示します。

1. まず、Notion で新しい内部インテグレーションをセットアップします: https://www.notion.so/profile/integrations
2. 新しい Notion キーをエクスポートします: `export NOTION_API_KEY=ntn_****`
3. 関連する Notion ページをインテグレーションに接続します。これを行うには、そのページにアクセスし、3つのドットをクリックして、「Connect to integration」を選択する必要があります。

依存関係: pip install agno mcp openai

使用法:
  python cookbook/tools/mcp/notion_mcp_agent.py
"""

import asyncio
import json
import os
from textwrap import dedent

from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.tools.mcp import MCPTools
from mcp import StdioServerParameters


async def run_agent():
    token = os.getenv("NOTION_API_KEY")
    if not token:
        raise ValueError(
            "Notion API キーが見つかりません: --NOTION_API_KEY を指定するか、NOTION_API_KEY 環境変数を設定してください"
        )

    command = "npx"
    args = ["-y", "@notionhq/notion-mcp-server"]
    env = {
        "OPENAPI_MCP_HEADERS": json.dumps(
            {"Authorization": f"Bearer {token}", "Notion-Version": "2022-06-28"}
        )
    }
    server_params = StdioServerParameters(command=command, args=args, env=env)

    async with MCPTools(server_params=server_params) as mcp_tools:
        agent = Agent(
            name="NotionDocsAgent",
            model=OpenAIChat(id="gpt-4o"),
            tools=[mcp_tools],
            description="MCP を介して Notion ドキュメントをクエリおよび変更するエージェント",
            instructions=dedent("""\
                MCP ツールを通じて Notion ドキュメントにアクセスできます。
                - ツールを使用してページを読み取り、検索、または更新します。
                - 変更を加える前にユーザーに確認してください。
            """),
            markdown=True,
            show_tool_calls=True,
        )

        await agent.acli_app(
            message="Notion ワークスペースやページにアクセスできる、役立つアシスタントです。",
            stream=True,
            markdown=True,
            exit_on=["exit", "quit"],
        )


if __name__ == "__main__":
    asyncio.run(run_agent())
