[Airbnb MCP server](https://github.com/openbnb-org/mcp-server-airbnb) を使用して、Airbnbリスティングを検索するエージェントを作成します。

```python
"""🏠 MCP Airbnb Agent - Airbnbリスティングを検索！

この例では、MCPとGemini 2.5 Proを使用してAirbnbリスティングを検索するエージェントを作成する方法を示します。

実行: 依存関係をインストールするには `pip install google-genai mcp agno` を実行してください。
"""

import asyncio

from agno.agent import Agent
from agno.models.google import Gemini
from agno.tools.mcp import MCPTools
from agno.utils.pprint import apprint_run_response


async def run_agent(message: str) -> None:
    async with MCPTools(
        "npx -y @openbnb/mcp-server-airbnb --ignore-robots-txt"
    ) as mcp_tools:
        agent = Agent(
            model=Gemini(id="gemini-2.5-pro-exp-03-25"),
            tools=[mcp_tools],
            markdown=True,
        )

        response_stream = await agent.arun(message, stream=True)
        await apprint_run_response(response_stream, markdown=True)


if __name__ == "__main__":
    asyncio.run(
        run_agent(
            "2025年8月1日から4日まで、サンフランシスコで2名で3泊利用可能なリスティングは何ですか？"
        )
    )
