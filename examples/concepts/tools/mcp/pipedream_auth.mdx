## コード

```python
"""
🔒 Pipedream MCP サーバーの認証付き利用

これは、Pipedream MCP サーバーを認証付きで利用する方法の例です。
これは、あなたのアプリがユーザーの代わりに MCP サーバーと連携する場合に役立ちます。

1. アクセストークンを取得します。取得方法は Pipedream のドキュメントで確認できます: https://pipedream.com/docs/connect/mcp/developers/
2. MCP サーバーの URL を取得します。次のような形式になります: https://remote.mcp.pipedream.net/<External user id>/<MCP app slug>
3. 環境変数を設定します:
    - MCP_SERVER_URL: 先ほど取得した MCP サーバーの URL
    - MCP_ACCESS_TOKEN: 先ほど取得したアクセストークン
    - PIPEDREAM_PROJECT_ID: 使用したい Pipedream プロジェクトのプロジェクト ID
    - PIPEDREAM_ENVIRONMENT: 使用したい Pipedream プロジェクトの環境
3. 依存関係をインストールします: pip install agno mcp-sdk
"""

import asyncio
from os import getenv

from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.tools.mcp import MCPTools, StreamableHTTPClientParams
from agno.utils.log import log_exception

mcp_server_url = getenv("MCP_SERVER_URL")
mcp_access_token = getenv("MCP_ACCESS_TOKEN")
pipedream_project_id = getenv("PIPEDREAM_PROJECT_ID")
pipedream_environment = getenv("PIPEDREAM_ENVIRONMENT")


server_params = StreamableHTTPClientParams(
    url=mcp_server_url,
    headers={
        "Authorization": f"Bearer {mcp_access_token}",
        "x-pd-project-id": pipedream_project_id,
        "x-pd-environment": pipedream_environment,
    },
)


async def run_agent(task: str) -> None:
    try:
        async with MCPTools(
            server_params=server_params, transport="streamable-http", timeout_seconds=20
        ) as mcp:
            agent = Agent(
                model=OpenAIChat(id="gpt-4o-mini"),
                tools=[mcp],
                markdown=True,
            )
            await agent.aprint_response(message=task, stream=True)
    except Exception as e:
        log_exception(f"予期せぬエラー: {e}")


if __name__ == "__main__":
    # エージェントはチャンネル、ユーザー、メッセージなどを読み取ることができます。
    asyncio.run(run_agent("Show me the latest message in the channel #general"))
```
