## 代码

```python cookbook/tools/google_maps_tools.py
from agno.agent import Agent
from agno.tools.google_maps import GoogleMapTools
from agno.tools.crawl4ai import Crawl4aiTools  # 可选：用于丰富地点数据

agent = Agent(
    name="地图 API 演示代理",
    tools=[
        GoogleMapTools(),
        Crawl4aiTools(max_length=5000),  # 可选：用于抓取企业网站
    ],
    description="专注于地图和基于位置的查询的地点和企业信息专家。",
    markdown=True,
    show_tool_calls=True,
)

# 示例 1：商家搜索
print("\n=== 商家搜索示例 ===")
agent.print_response(
    "帮我查找菲尼克斯（Phoenix, AZ）评价很高的印度餐厅及其联系方式",
    markdown=True,
    stream=True,
)

# 示例 2：路线规划
print("\n=== 路线规划示例 ===")
agent.print_response(
    """获取从“菲尼克斯天港机场”（Phoenix Sky Harbor Airport）到“沙漠植物园”（Desert Botanical Garden）的驾车路线，
    如果可能的话，避开高速公路。""",
    markdown=True,
    stream=True,
)

# 示例 3：地址验证和地理编码
print("\n=== 地址验证和地理编码示例 ===")
agent.print_response(
    """请验证并地理编码此地址：
    '1600 Amphitheatre Parkway, Mountain View, CA'""",
    markdown=True,
    stream=True,
)

# 示例 4：距离矩阵
print("\n=== 距离矩阵示例 ===")
agent.print_response(
    """计算菲尼克斯（Phoenix）以下地点之间的出行时间和距离：
    起点：['菲尼克斯天港机场', '菲尼克斯市中心']
    终点：['沙漠植物园', '菲尼克斯动物园']""",
    markdown=True,
    stream=True,
)

# 示例 5：地点分析
print("\n=== 地点分析示例 ===")
agent.print_response(
    """分析菲尼克斯（Phoenix）的这个地点：
    地址：'2301 N Central Ave, Phoenix, AZ 85004'
    请提供：
    1. 精确坐标
    2. 附近的标志性建筑
    3. 海拔数据
    4. 当地时区""",
    markdown=True,
    stream=True,
)

# 示例 6：多模式交通方式比较
print("\n=== 交通方式示例 ===")
agent.print_response(
    """比较从“菲尼克斯会议中心”（Phoenix Convention Center）到“菲尼克斯艺术博物馆”（Phoenix Art Museum）的各种出行模式：
    1. 驾车
    2. 步行
    3. 公共交通（如果可用）
    包含每种选项的预估时间和距离。""",
    markdown=True,
    stream=True,
)
```

  <Step title="设置您的 API 密钥">
    ```bash
    export GOOGLE_MAPS_API_KEY=xxx
    export OPENAI_API_KEY=xxx
    ```
    从 [Google Cloud Console](https://console.cloud.google.com/projectselector2/google/maps-apis/credentials) 获取您的 API 密钥
  </Step>

  <Step title="安装库">
    ```bash
    pip install -U openai googlemaps agno
    ```
  </Step>

  <Step title="运行代理">
    <CodeGroup>
    ```bash Mac
    python cookbook/tools/google_maps_tools.py
    ```

    ```bash Windows
    python cookbook/tools/google_maps_tools.py
    ```
    </CodeGroup>
  </Step>
</Steps>
