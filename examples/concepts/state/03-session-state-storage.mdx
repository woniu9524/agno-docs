---
title: Storage を使用した永続的な状態
---
この例では、SQLite ストレージを使用してエージェントのセッション状態を永続化し、複数の実行にわたって継続性を持たせる方法を示します。
## コード

```python cookbook/agent_concepts/state/session_state_storage.py

from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.storage.sqlite import SqliteStorage


# ショッピングリストにアイテムを追加するツールを定義します
def add_item(agent: Agent, item: str) -> str:
    """ショッピングリストにアイテムを追加します。"""
    if item not in agent.session_state["shopping_list"]:
        agent.session_state["shopping_list"].append(item)
    return f"ショッピングリストは現在 {agent.session_state['shopping_list']} です"


agent = Agent(
    model=OpenAIChat(id="gpt-4o-mini"),
    # 実行サイクル間で同じセッションを継続するためにセッションIDを固定します
    session_id="fixed_id_for_demo",
    # セッション状態を空のショッピングリストで初期化します
    session_state={"shopping_list": []},
    # ショッピングリストにアイテムを追加するツールを追加します
    tools=[add_item],
    # セッション状態をSQLiteデータベースに保存します
    storage=SqliteStorage(table_name="agent_sessions", db_file="tmp/data.db"),
    # 現在のショッピングリストを状態から指示に追加します
    instructions="現在のショッピングリストは: {shopping_list}",
    # 重要: `add_state_in_messages=True` を設定します
    # これにより、指示内で `{shopping_list}` が利用可能になります
    add_state_in_messages=True,
    markdown=True,
)

# 使用例
agent.print_response("私のショッピングリストは何ですか？", stream=True)
print(f"セッション状態: {agent.session_state}")
agent.print_response("牛乳、卵、パンを追加してください", stream=True)
print(f"セッション状態: {agent.session_state}")
```
