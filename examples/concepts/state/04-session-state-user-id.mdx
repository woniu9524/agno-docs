title: マルチユーザーの状態管理
---
この例では、マルチユーザー環境で各ユーザーの状態を維持する方法を示します。
## コード

```python cookbook/agent_concepts/state/session_state_user_id.py
import json

from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.utils.log import log_info

# ユーザーの買い物リストを保存するインメモリデータベース
# ユーザーIDとセッションIDで整理されます
shopping_list = {}


def add_item(agent: Agent, item: str) -> str:
    """現在のユーザーの買い物リストにアイテムを追加します。"""
    current_user_id = agent.session_state["current_user_id"]
    current_session_id = agent.session_state["current_session_id"]
    shopping_list.setdefault(current_user_id, {}).setdefault(
        current_session_id, []
    ).append(item)
    return f"アイテム {item} を買い物リストに追加しました"


def remove_item(agent: Agent, item: str) -> str:
    """現在のユーザーの買い物リストからアイテムを削除します。"""
    current_user_id = agent.session_state["current_user_id"]
    current_session_id = agent.session_state["current_session_id"]

    if (
        current_user_id not in shopping_list
        or current_session_id not in shopping_list[current_user_id]
    ):
        return f"ユーザー {current_user_id} およびセッション {current_session_id} の買い物リストが見つかりません"

    if item not in shopping_list[current_user_id][current_session_id]:
        return f"アイテム '{item}' は、ユーザー {current_user_id} およびセッション {current_session_id} の買い物リストに見つかりません"

    shopping_list[current_user_id][current_session_id].remove(item)
    return f"アイテム {item} を買い物リストから削除しました"


def get_shopping_list(agent: Agent) -> str:
    """現在のユーザーの買い物リストを取得します。"""
    current_user_id = agent.session_state["current_user_id"]
    current_session_id = agent.session_state["current_session_id"]
    return f"ユーザー {current_user_id} およびセッション {current_session_id} の買い物リスト: \n{json.dumps(shopping_list[current_user_id][current_session_id], indent=2)}"


# 状態を維持するAgentを作成します
agent = Agent(
    model=OpenAIChat(id="gpt-4o-mini"),
    tools=[add_item, remove_item, get_shopping_list],
    # インメモリデータベースを参照します
    instructions=[
        "現在のユーザーID: {current_user_id}",
        "現在のセッションID: {current_session_id}",
    ],
    # 重要: インストラクションに状態を追加します
    add_state_in_messages=True,
    markdown=True,
)

user_id_1 = "john_doe"
user_id_2 = "mark_smith"
user_id_3 = "carmen_sandiago"

# 使用例
agent.print_response(
    "牛乳、卵、パンを買い物リストに追加してください",
    stream=True,
    user_id=user_id_1,
    session_id="user_1_session_1",
)
agent.print_response(
    "タコスを買い物リストに追加してください",
    stream=True,
    user_id=user_id_2,
    session_id="user_2_session_1",
)
agent.print_response(
    "りんごとぶどうを買い物リストに追加してください",
    stream=True,
    user_id=user_id_3,
    session_id="user_3_session_1",
)
agent.print_response(
    "牛乳を買い物リストから削除してください",
    stream=True,
    user_id=user_id_1,
    session_id="user_1_session_1",
)
agent.print_response(
    "ひき肉を買い物リストに追加してください",
    stream=True,
    user_id=user_id_2,
    session_id="user_2_session_1",
)

# マーク・スミスの買い物リストには何が入っていますか？
agent.print_response(
    "マーク・スミスの買い物リストには何が入っていますか？",
    stream=True,
    user_id=user_id_2,
    session_id="user_2_session_1",
)

# 新しいセッションなので、買い物リストも新しいものになります
agent.print_response(
    "鶏肉とスープを私のリストに追加してください。",
    stream=True,
    user_id=user_id_2,
    session_id="user_3_session_2",
)

print(f"最終的な買い物リスト: \n{json.dumps(shopping_list, indent=2)}")
```
