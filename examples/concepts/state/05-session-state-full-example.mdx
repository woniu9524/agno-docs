title: 複数回の実行にわたる状態管理

この例では、複数回の実行にわたって状態を管理できるステートフルなエージェントの構築方法を説明します。
## コード

```python cookbook/agent_concepts/state/shopping_list.py
from textwrap import dedent

from agno.agent import Agent
from agno.models.openai import OpenAIChat


# 쇼핑 목록을 관리하는 도구 정의
def add_item(agent: Agent, item: str) -> str:
    """쇼핑 목록에 항목을 추가하고 확인 메시지를 반환합니다."""
    # 항목이 이미 목록에 없으면 추가합니다.
    if item.lower() not in [i.lower() for i in agent.session_state["shopping_list"]]:
        agent.session_state["shopping_list"].append(item)
        return f"'{item}'을(를) 쇼핑 목록에 추가했습니다."
    else:
        return f"'{item}'은(는) 이미 쇼핑 목록에 있습니다."


def remove_item(agent: Agent, item: str) -> str:
    """이름으로 쇼핑 목록에서 항목을 제거합니다."""
    # 대소문자를 구분하지 않는 검색
    for i, list_item in enumerate(agent.session_state["shopping_list"]):
        if list_item.lower() == item.lower():
            agent.session_state["shopping_list"].pop(i)
            return f"'{list_item}'을(를) 쇼핑 목록에서 제거했습니다."

    return f"'{item}'을(를) 쇼핑 목록에서 찾을 수 없습니다."


def list_items(agent: Agent) -> str:
    """쇼핑 목록의 모든 항목을 나열합니다."""
    shopping_list = agent.session_state["shopping_list"]

    if not shopping_list:
        return "쇼핑 목록이 비어 있습니다."

    items_text = "\n".join([f"- {item}" for item in shopping_list])
    return f"현재 쇼핑 목록:\n{items_text}"


# 상태를 유지하는 쇼핑 목록 관리자 에이전트 생성
agent = Agent(
    model=OpenAIChat(id="gpt-4o-mini"),
    # 세션 상태를 빈 쇼핑 목록으로 초기화합니다.
    session_state={"shopping_list": []},
    tools=[add_item, remove_item, list_items],
    # 세션 상태의 변수를 지침에 사용할 수 있습니다.
    instructions=dedent("""\
        당신의 임무는 쇼핑 목록을 관리하는 것입니다.

        쇼핑 목록은 비어 있는 상태에서 시작합니다. 항목을 추가하거나, 이름으로 항목을 제거하거나, 모든 항목을 나열할 수 있습니다.

        현재 쇼핑 목록: {shopping_list}
    """),
    add_state_in_messages=True,
    markdown=True,
)

# 사용 예시
agent.print_response("우유, 계란, 빵을 쇼핑 목록에 추가해 줘", stream=True)
print(f"세션 상태: {agent.session_state}")

agent.print_response("빵을 샀어", stream=True)
print(f"세션 상태: {agent.session_state}")

agent.print_response("사과와 오렌지가 필요해", stream=True)
print(f"세션 상태: {agent.session_state}")

agent.print_response("내 목록에 뭐가 있지?", stream=True)
print(f"세션 상태: {agent.session_state}")

agent.print_response(
    "내 목록에서 모든 것을 지우고 바나나와 요거트만으로 다시 시작해 줘",
    stream=True,
)
print(f"세션 상태: {agent.session_state}")


```
