## コード

```python cookbook/agent_concepts/user_control_flows/external_tool_execution_stream.py
import subprocess

from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.tools import tool
from agno.utils import pprint

@tool(external_execution=True)
def execute_shell_command(command: str) -> str:
    """シェルコマンドを実行します。

    Args:
        command (str): 実行するシェルコマンド

    Returns:
        str: シェルコマンドの出力
    """
    if command.startswith("ls"):
        return subprocess.check_output(command, shell=True).decode("utf-8")
    else:
        raise Exception(f"サポートされていないコマンド: {command}")

agent = Agent(
    model=OpenAIChat(id="gpt-4o-mini"),
    tools=[execute_shell_command],
    markdown=True,
)

for run_response in agent.run(
    "現在のディレクトリにあるファイルは何ですか？", stream=True
):
    if run_response.is_paused:
        for tool in run_response.tools_awaiting_external_execution:
            if tool.tool_name == execute_shell_command.name:
                print(f"{tool.tool_name} を引数 {tool.tool_args} で外部実行します")
                # ツールを自分で実行します。完全に外部のものを実行することもできます。
                result = execute_shell_command.entrypoint(**tool.tool_args)
                # エージェントが続行できるように、ツール実行オブジェクトに結果を設定する必要があります
                tool.result = result

        run_response = agent.continue_run(run_response=run_response, stream=True)
        pprint.pprint_run_response(run_response)
```

## 主な機能

- ストリーミング応答のために `agent.run(stream=True)` を使用します
- `agent.continue_run(stream=True)` でストリーミング継続を実装します
- 外部ツール実行によるリアルタイムの対話を維持します
- 外部ツールでのストリーミング応答の処理方法を示します
