## コード

```python cookbook/agent_concepts/user_control_flows/external_tool_execution.py
import subprocess

from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.tools import tool
from agno.utils import pprint

@tool(external_execution=True)
def execute_shell_command(command: str) -> str:
    """シェルコマンドを実行します。

    Args:
        command (str): 実行するシェルコマンド

    Returns:
        str: シェルコマンドの出力
    """
    if command.startswith("ls"):
        return subprocess.check_output(command, shell=True).decode("utf-8")
    else:
        raise Exception(f"Unsupported command: {command}")

agent = Agent(
    model=OpenAIChat(id="gpt-4o-mini"),
    tools=[execute_shell_command],
    markdown=True,
)

run_response = agent.run("現在のディレクトリにあるファイルは何ですか？")
if run_response.is_paused:
    for tool in run_response.tools_awaiting_external_execution:
        if tool.tool_name == execute_shell_command.name:
            print(f"Executing {tool.tool_name} with args {tool.tool_args} externally")
            # ツールを自分で実行します。完全に外部で何かを実行することもできます。
            result = execute_shell_command.entrypoint(**tool.tool_args)
            # エージェントが続行できるように、ツール実行オブジェクトに結果を設定する必要があります
            tool.result = result

    run_response = agent.continue_run()
    pprint.pprint_run_response(run_response)
```

## 主な機能

- 外部実行が必要なツールをマークするために `@tool(external_execution=True)` を使用します
- ツール実行結果の処理方法を示します
