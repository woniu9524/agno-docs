## コード

```python cookbook/agent_concepts/user_control_flows/user_input_required_async.py
import asyncio
from typing import List

from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.tools import tool
from agno.tools.function import UserInputField
from agno.utils import pprint

@tool(requires_user_input=True, user_input_fields=["to_address"])
async def send_email(subject: str, body: str, to_address: str) -> str:
    """
    メールを送信します。

    Args:
        subject (str): メールの件名。
        body (str): メールの本文。
        to_address (str): 送信先のメールアドレス。
    """
    return f"Sent email to {to_address} with subject {subject} and body {body}"

agent = Agent(
    model=OpenAIChat(id="gpt-4o-mini"),
    tools=[send_email],
    markdown=True,
)

asyncio.run(
    agent.arun("Send an email with the subject 'Hello' and the body 'Hello, world!'")
)
if agent.is_paused:
    for tool in agent.run_response.tools_requiring_user_input:
        input_schema: List[UserInputField] = tool.user_input_schema

        for field in input_schema:
            # ユーザーにフィールド情報を表示します
            print(f"\nフィールド: {field.name}")
            print(f"説明: {field.description}")
            print(f"タイプ: {field.field_type}")

            # ユーザー入力を取得します
            if field.value is None:
                user_value = input(f"{field.name} の値を入力してください: ")
                # フィールドの値を更新します
                field.value = user_value
            else:
                print(f"値: {field.value}")


    run_response = asyncio.run(agent.acontinue_run())
    pprint.pprint_run_response(run_response)
```

## 主な機能

- 非同期エージェント実行のために `agent.arun()` を使用します
- 非同期継続のために `agent.acontinue_run()` を実装します
- 同期バージョンと同じユーザー入力フローを維持します
- ユーザー入力収集による非同期実行の処理方法を示します
