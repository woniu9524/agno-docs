---

## コード

```python
from agno.agent import Agent
from agno.knowledge.pdf_url import PDFUrlKnowledgeBase
from agno.vectordb.lancedb import LanceDb

# LanceDB の初期化
# デフォルトでは、データは /tmp/lancedb に保存されます
vector_db = LanceDb(
    table_name="recipes",
    uri="tmp/lancedb",  # データを別の場所に保存するには、このパスを変更できます
)

# ステップ 1: ドキュメントとメタデータで知識ベースを初期化
# ------------------------------------------------------------------------------
# 知識ベースを初期化する際に、フィルタリングに使用されるメタデータを添付できます
# このメタデータには、ユーザー ID、ドキュメントの種類、日付、またはその他の属性を含めることができます

knowledge_base = PDFUrlKnowledgeBase(
    urls=[
        {
            "url": "https://agno-public.s3.amazonaws.com/recipes/thai_recipes_short.pdf",
            "metadata": {
                "cuisine": "Thai",
                "source": "Thai Cookbook",
                "region": "Southeast Asia",
            },
        },
        {
            "url": "https://agno-public.s3.amazonaws.com/recipes/cape_recipes_short_2.pdf",
            "metadata": {
                "cuisine": "Cape",
                "source": "Cape Cookbook",
                "region": "South Africa",
            },
        },
    ],
    vector_db=vector_db,
)

# すべてのドキュメントをベクトルデータベースにロードします
knowledge_base.load(recreate=True)

# ステップ 2: クエリからのフィルタを使用して Agent で知識ベースをクエリ
# -----------------------------------------------------------------------------------

# エージェントによるフィルタリングを有効にします
agent = Agent(
    knowledge=knowledge_base,
    search_knowledge=True,
    enable_agentic_knowledge_filters=True,
)

# クエリにフィルタを含めることで、Agent が自動的にそれらを拾えるようにして、
# Jordan Mitchell の経験とスキルについてクエリします
agent.print_response(
    "Pad Thai の作り方を教えてください。cuisine が Thai で source が Thai Cookbook のドキュメントを参照してください",
    markdown=True,
)
```
