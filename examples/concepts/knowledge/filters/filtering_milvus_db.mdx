---

## コード

```python
from agno.agent import Agent
from agno.knowledge.pdf import PDFKnowledgeBase
from agno.utils.media import (
    SampleDataFileExtension,
    download_knowledge_filters_sample_data,
)
from agno.vectordb.milvus import Milvus

# すべてのサンプルCVをダウンロードし、そのパスを取得します
downloaded_cv_paths = download_knowledge_filters_sample_data(
    num_files=5, file_extension=SampleDataFileExtension.PDF
)

# MilvusベクトルDBを初期化します
vector_db = Milvus(
    collection="recipes",
    uri="tmp/milvus.db",
)

# ステップ1: ドキュメントとメタデータでナレッジベースを初期化します
# ------------------------------------------------------------------------------
# ナレッジベースを初期化する際に、フィルタリングに使用されるメタデータを添付できます。
# このメタデータには、ユーザーID、ドキュメントタイプ、日付、またはその他の属性を含めることができます。

knowledge_base = PDFKnowledgeBase(
    path=[
        {
            "path": downloaded_cv_paths[0],
            "metadata": {
                "user_id": "jordan_mitchell",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[1],
            "metadata": {
                "user_id": "taylor_brooks",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[2],
            "metadata": {
                "user_id": "morgan_lee",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[3],
            "metadata": {
                "user_id": "casey_jordan",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[4],
            "metadata": {
                "user_id": "alex_rivera",
                "document_type": "cv",
                "year": 2025,
            },
        },
    ],
    vector_db=vector_db,
)

# すべてのドキュメントをベクトルデータベースにロードします
knowledge_base.load(recreate=True)

# ステップ2: さまざまなフィルタの組み合わせでナレッジベースをクエリします
# ------------------------------------------------------------------------------

agent = Agent(
    knowledge=knowledge_base,
    search_knowledge=True,
)

agent.print_response(
    "Jordan Mitchellの経験とスキルについて教えてください",
    knowledge_filters={"user_id": "jordan_mitchell"},
    markdown=True,
)
```
