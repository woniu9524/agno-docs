## コード

```python
from agno.agent import Agent
from agno.knowledge.text import TextKnowledgeBase
from agno.utils.media import (
    SampleDataFileExtension,
    download_knowledge_filters_sample_data,
)
from agno.vectordb.lancedb import LanceDb

# すべてのサンプルCVをダウンロードし、そのパスを取得します
downloaded_cv_paths = download_knowledge_filters_sample_data(
    num_files=5, file_extension=SampleDataFileExtension.TXT
)

# LanceDBを初期化します
# デフォルトでは、データは /tmp/lancedb に保存されます
vector_db = LanceDb(
    table_name="recipes",
    uri="tmp/lancedb",  # データを別の場所に保存するには、このパスを変更できます
)

# ステップ1: ドキュメントとメタデータでナレッジベースを初期化します
# ------------------------------------------------------------------------------
# ナレッジベースをロードする際に、フィルタリングに使用されるメタデータをアタッチできます
# このメタデータには、ユーザーID、ドキュメントタイプ、日付、またはその他の属性を含めることができます

# TextKnowledgeBaseを初期化します
knowledge_base = TextKnowledgeBase(
    vector_db=vector_db,
    num_documents=5,
)

knowledge_base.load_document(
    path=downloaded_cv_paths[0],
    metadata={"user_id": "jordan_mitchell", "document_type": "cv", "year": 2025},
    recreate=True,  # 最初の実行時のみTrueに設定し、その後はFalseに設定してください
)

# user_2メタデータで2番目のドキュメントをロードします
knowledge_base.load_document(
    path=downloaded_cv_paths[1],
    metadata={"user_id": "taylor_brooks", "document_type": "cv", "year": 2025},
)

# user_3メタデータで2番目のドキュメントをロードします
knowledge_base.load_document(
    path=downloaded_cv_paths[2],
    metadata={"user_id": "morgan_lee", "document_type": "cv", "year": 2025},
)

# user_4メタデータで2番目のドキュメントをロードします
knowledge_base.load_document(
    path=downloaded_cv_paths[3],
    metadata={"user_id": "casey_jordan", "document_type": "cv", "year": 2025},
)

# user_5メタデータで2番目のドキュメントをロードします
knowledge_base.load_document(
    path=downloaded_cv_paths[4],
    metadata={"user_id": "alex_rivera", "document_type": "cv", "year": 2025},
)

# ステップ2: 異なるフィルタの組み合わせでナレッジベースをクエリします
# ------------------------------------------------------------------------------
# 実行したい例のコメントを解除してください

# オプション1: エージェントでのフィルタリング
# ナレッジベースでエージェントを初期化します
agent = Agent(
    knowledge=knowledge_base,
    search_knowledge=True,
    knowledge_filters={"user_id": "jordan_mitchell"},
)
agent.print_response(
    "Jordan Mitchellの経験とスキルについて教えてください",
    markdown=True,
)

```
