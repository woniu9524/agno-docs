## コード

```python
"""
ユーザーレベルのナレッジフィルタリングの例

このクックブックでは、ナレッジフィルタを使用して、特定のユーザー、ドキュメントタイプ、またはその他のメタデータ属性にナレッジベースの検索を制限する方法を示します。

実証される主な概念：
1. ユーザー固有のメタデータを持つドキュメントのロード
2. ユーザーIDによるナレッジベース検索のフィルタリング
3. 複数のフィルタ条件の組み合わせ
4. 異なるフィルタの組み合わせ間での結果の比較

以下の方法でフィルタを渡すことができます：
1. Agentにのみ渡す場合、すべての実行に使用します
2. run/print_responseにのみ渡す場合、その実行に使用します
3. 両方に渡す場合、その実行のためにrun/print_responseに渡されたフィルタで上書きします
"""

from agno.agent import Agent
from agno.knowledge.text import TextKnowledgeBase
from agno.utils.media import (
    SampleDataFileExtension,
    download_knowledge_filters_sample_data,
)
from agno.vectordb.lancedb import LanceDb

# すべてのサンプルCVをダウンロードし、そのパスを取得します
downloaded_cv_paths = download_knowledge_filters_sample_data(
    num_files=5, file_extension=SampleDataFileExtension.TXT
)

# LanceDBを初期化します
# デフォルトでは、データは /tmp/lancedb に保存されます
vector_db = LanceDb(
    table_name="recipes",
    uri="tmp/lancedb",  # データを別の場所に保存するには、このパスを変更できます
)

# ステップ1：ドキュメントとメタデータでナレッジベースを初期化します
# ------------------------------------------------------------------------------
# ナレッジベースを初期化する際に、フィルタリングに使用されるメタデータをアタッチできます
# このメタデータには、ユーザーID、ドキュメントタイプ、日付、またはその他の属性を含めることができます

knowledge_base = TextKnowledgeBase(
    path=[
        {
            "path": downloaded_cv_paths[0],
            "metadata": {
                "user_id": "jordan_mitchell",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[1],
            "metadata": {
                "user_id": "taylor_brooks",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[2],
            "metadata": {
                "user_id": "morgan_lee",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[3],
            "metadata": {
                "user_id": "casey_jordan",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[4],
            "metadata": {
                "user_id": "alex_rivera",
                "document_type": "cv",
                "year": 2025,
            },
        },
    ],
    vector_db=vector_db,
)

# すべてのドキュメントをベクトルデータベースにロードします
knowledge_base.load(recreate=True)

# ステップ2：さまざまなフィルタの組み合わせでナレッジベースをクエリします
# ------------------------------------------------------------------------------

# オプション1：Agentでのフィルタリング
# ナレッジベースとフィルタでAgentを初期化します
agent = Agent(
    knowledge=knowledge_base,
    search_knowledge=True,
    knowledge_filters={"user_id": "jordan_mitchell"},
)

# Jordan Mitchellの経験とスキルについてクエリします
agent.print_response(
    "Tell me about Jordan Mitchell's experience and skills",
    markdown=True,
)

# # オプション2：run/print_responseでのフィルタリング
# agent = Agent(
#     knowledge=knowledge_base,
#     search_knowledge=True,
# )

# # Taylor Brooksについて候補者としてクエリします
# agent.print_response(
#     "Tell me about Taylor Brooks as a candidate",
#     knowledge_filters={"user_id": "taylor_brooks"},
#     markdown=True,
# )
```
