---

## コード

```python
from pathlib import Path

from agno.agent import Agent
from agno.knowledge.pdf import PDFKnowledgeBase
from agno.utils.media import (
    SampleDataFileExtension,
    download_knowledge_filters_sample_data,
)
from agno.vectordb.lancedb import LanceDb

# すべてのサンプル CV をダウンロードし、そのパスを取得します
downloaded_cv_paths = download_knowledge_filters_sample_data(
    num_files=5, file_extension=SampleDataFileExtension.PDF
)

# LanceDB を初期化します
# デフォルトでは、データは /tmp/lancedb に保存されます
vector_db = LanceDb(
    table_name="recipes",
    uri="tmp/lancedb",  # データを別の場所に保存するには、このパスを変更できます
)

# ステップ 1: ドキュメントとメタデータで知識ベースを初期化します
# ------------------------------------------------------------------------------
# 知識ベースを初期化する際に、フィルタリングに使用されるメタデータを添付できます
# このメタデータには、ユーザー ID、ドキュメントの種類、日付、またはその他の属性を含めることができます

knowledge_base = PDFKnowledgeBase(
    path=[
        {
            "path": downloaded_cv_paths[0],
            "metadata": {
                "user_id": "jordan_mitchell",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[1],
            "metadata": {
                "user_id": "taylor_brooks",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[2],
            "metadata": {
                "user_id": "morgan_lee",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[3],
            "metadata": {
                "user_id": "casey_jordan",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[4],
            "metadata": {
                "user_id": "alex_rivera",
                "document_type": "cv",
                "year": 2025,
            },
        },
    ],
    vector_db=vector_db,
)

# すべてのドキュメントをベクトルデータベースにロードします
knowledge_base.load(recreate=True)

# ステップ 2: クエリからのフィルタを使用してエージェントで知識ベースをクエリします
# -----------------------------------------------------------------------------------

# エージェントによるフィルタリングを有効にします
agent = Agent(
    knowledge=knowledge_base,
    search_knowledge=True,
    enable_agentic_knowledge_filters=True,
)

# Jordan Mitchell の経験とスキルについて、ユーザー ID を jordan_mitchell、ドキュメントの種類を cv としてクエリします
agent.print_response(
    "Tell me about Jordan Mitchell's experience and skills with jordan_mitchell as user id and document type cv",
    markdown=True,
)
```
