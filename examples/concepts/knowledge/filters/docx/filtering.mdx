## コード

```python
"""
ユーザーレベルのナレッジフィルタリングの例

このクックブックでは、ナレッジフィルタを使用して、ナレッジベース検索を特定のユーザー、ドキュメントタイプ、またはその他のメタデータ属性に制限する方法を示します。

実証されている主要な概念:
1. ユーザー固有のメタデータを持つドキュメントのロード
2. ユーザー ID によるナレッジベース検索のフィルタリング
3. 複数のフィルタ条件の組み合わせ
4. 異なるフィルタの組み合わせ間での結果の比較

フィルタは次の方法で渡すことができます:
1. Agent のみに渡す場合、すべての実行に使用します
2. run/print_response のみに渡す場合、その実行に使用します
3. 両方に渡す場合、その実行のために run/print_response で渡されたフィルタでオーバーライドします
"""

from agno.agent import Agent
from agno.knowledge.docx import DocxKnowledgeBase
from agno.utils.media import (
    SampleDataFileExtension,
    download_knowledge_filters_sample_data,
)
from agno.vectordb.lancedb import LanceDb

# すべてのサンプル CV をダウンロードし、そのパスを取得します
downloaded_cv_paths = download_knowledge_filters_sample_data(
    num_files=5, file_extension=SampleDataFileExtension.DOCX
)

# LanceDB の初期化
vector_db = LanceDb(
    table_name="recipes",
    uri="tmp/lancedb",
)

# ダウンロードされたパスをナレッジベースの初期化で使用します
knowledge_base = DocxKnowledgeBase(
    path=[
        {
            "path": downloaded_cv_paths[0],
            "metadata": {
                "user_id": "jordan_mitchell",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[1],
            "metadata": {
                "user_id": "taylor_brooks",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[2],
            "metadata": {
                "user_id": "morgan_lee",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[3],
            "metadata": {
                "user_id": "casey_jordan",
                "document_type": "cv",
                "year": 2025,
            },
        },
        {
            "path": downloaded_cv_paths[4],
            "metadata": {
                "user_id": "alex_rivera",
                "document_type": "cv",
                "year": 2025,
            },
        },
    ],
    vector_db=vector_db,
)


# すべてのドキュメントをベクトルデータベースにロードします
knowledge_base.load(recreate=True)

# ステップ 2: 異なるフィルタの組み合わせでナレッジベースにクエリを実行します
# ------------------------------------------------------------------------------

# オプション 1: Agent でのフィルタリング
# ナレッジベースとフィルタで Agent を初期化します
agent = Agent(
    knowledge=knowledge_base,
    search_knowledge=True,
    knowledge_filters={"user_id": "jordan_mitchell"},
)

# Jordan Mitchell の経験とスキルについてクエリを実行します
agent.print_response(
    "Tell me about Jordan Mitchell's experience and skills",
    markdown=True,
)

# # オプション 2: run/print_response でのフィルタリング
# agent = Agent(
#     knowledge=knowledge_base,
#     search_knowledge=True,
# )

# # Taylor Brooks を候補者としてクエリを実行します
# agent.print_response(
#     "Tell me about Taylor Brooks as a candidate",
#     knowledge_filters={"user_id": "taylor_brooks"},
#     markdown=True,
# )
```
