## コード

```python
from agno.agent import Agent
from agno.embedder.openai import OpenAIEmbedder
from agno.knowledge.pdf_url import PDFUrlKnowledgeBase
from agno.models.openai import OpenAIChat
from agno.vectordb.lancedb import LanceDb, SearchType

# URL から PDF の知識ベースを作成します
knowledge_base = PDFUrlKnowledgeBase(
    urls=["https://agno-public.s3.amazonaws.com/recipes/ThaiRecipes.pdf"],
    # LanceDB をベクトルデータベースとして使用し、埋め込みを `recipes` テーブルに保存します
    vector_db=LanceDb(
        table_name="recipes",
        uri="tmp/lancedb",
        search_type=SearchType.vector,
        embedder=OpenAIEmbedder(id="text-embedding-3-small"),
    ),
)
# 知識ベースをロードします: 知識ベースは既にロードされているため、初回実行後にコメントアウトしてください
knowledge_base.load()

agent = Agent(
    model=OpenAIChat(id="gpt-4o"),
    knowledge=knowledge_base,
    # AgentKnowledge からの参照をユーザープロンプトに追加することで、RAG を有効にします。
    add_references=True,
    # Agent はデフォルトで `search_knowledge=True` になるため、False に設定します
    search_knowledge=False,
    show_tool_calls=True,
    markdown=True,
)
agent.print_response(
    "ココナッツミルクのスープでチキンとガランガルを作る方法", stream=True
)
```
