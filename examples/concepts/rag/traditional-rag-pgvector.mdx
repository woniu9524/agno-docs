## コード

```python
from agno.agent import Agent
from agno.embedder.openai import OpenAIEmbedder
from agno.knowledge.pdf_url import PDFUrlKnowledgeBase
from agno.models.openai import OpenAIChat
from agno.vectordb.pgvector import PgVector, SearchType

db_url = "postgresql+psycopg://ai:ai@localhost:5532/ai"
# URL から PDF のナレッジベースを作成
knowledge_base = PDFUrlKnowledgeBase(
    urls=["https://agno-public.s3.amazonaws.com/recipes/ThaiRecipes.pdf"],
    # PgVector をベクトルデータベースとして使用し、`ai.recipes` テーブルに埋め込みを保存
    vector_db=PgVector(
        table_name="recipes",
        db_url=db_url,
        search_type=SearchType.hybrid,
        embedder=OpenAIEmbedder(id="text-embedding-3-small"),
    ),
)
# ナレッジベースをロード: 初回実行後にコメントアウトしてください。ナレッジベースは既にロードされています。
knowledge_base.load(upsert=True)

agent = Agent(
    model=OpenAIChat(id="gpt-4o"),
    knowledge=knowledge_base,
    # `knowledge` からのコンテキストをユーザープロンプトに追加することで RAG を有効化
    add_references=True,
    # Agents はデフォルトで `search_knowledge=True` になるため False に設定
    search_knowledge=False,
    markdown=True,
)
agent.print_response(
    "ココナッツミルクとガランガルのチキンスープの作り方を教えてください", stream=True
)
```
