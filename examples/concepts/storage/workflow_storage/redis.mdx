Agno は、`RedisStorage` クラスを使用して、ワークフローのストレージバックエンドとして Redis をサポートしています。

## 使用方法

### Redis の実行

[docker desktop](https://docs.docker.com/desktop/install/mac-install/) をインストールし、次のコマンドでポート **6379** で **Redis** を実行します。

```bash
docker run --name my-redis -p 6379:6379 -d redis
```

```python redis_storage_for_workflow.py
"""
実行方法: `pip install openai httpx newspaper4k redis agno` で依存関係をインストールします。
"""

import json
from typing import Iterator

import httpx
from agno.agent import Agent
from agno.run.response import RunResponse
from agno.storage.redis import RedisStorage
from agno.tools.newspaper4k import Newspaper4kTools
from agno.utils.log import logger
from agno.utils.pprint import pprint_run_response
from agno.workflow import Workflow

# デフォルトのローカル接続で Redis ストレージを初期化します
storage = RedisStorage(
    # セッションの名前空間を設定するための Redis キーのプレフィックス
    prefix="agno_test",
    # Redis ホストアドレス
    host="localhost",
    # Redis ポート番号
    port=6379,
)


class HackerNewsReporter(Workflow):
    description: str = (
        "Hacker News のトップストーリーを取得し、それらに関するレポートを作成します。"
    )

    hn_agent: Agent = Agent(
        description="Hacker News のトップストーリーを取得します。 "
        "URL、スコア、タイトル、および利用可能な場合は要約を含む、すべての可能な情報を共有します。",
        show_tool_calls=True,
    )

    writer: Agent = Agent(
        tools=[Newspaper4kTools()],
        description="Hacker News のトップストーリーに関する魅力的なレポートを作成します。",
        instructions=[
            "トップストーリーとそのリンクが提供されます。",
            "各記事を注意深く読み、内容について考えてください。",
            "その後、ニューヨーク・タイムズ紙にふさわしい最終記事を生成してください。",
            "記事をセクションに分割し、最後に主要なポイントを提供してください。",
            "タイトルはキャッチーで魅力的なものにしてください。",
            "各記事のスコア、タイトル、URL、要約を共有してください。",
            "セクションに適切なタイトルを付け、各セクションに詳細/事実/プロセスを提供してください。",
            "読み取れない、または理解できない記事は無視してください。",
            "REMEMBER: ニューヨーク・タイムズ紙向けの記事を作成しているので、記事の品質が重要です。",
        ],
    )

    def get_top_hackernews_stories(self, num_stories: int = 10) -> str:
        """Hacker News のトップストーリーを取得するためにこの関数を使用します。

        Args:
            num_stories (int): 返されるストーリーの数。デフォルトは 10 です。

        Returns:
            str: トップストーリーの JSON 文字列。
        """

        # トップストーリー ID を取得します
        response = httpx.get("https://hacker-news.firebaseio.com/v0/topstories.json")
        story_ids = response.json()

        # ストーリーの詳細を取得します
        stories = []
        for story_id in story_ids[:num_stories]:
            story_response = httpx.get(
                f"https://hacker-news.firebaseio.com/v0/item/{story_id}.json"
            )
            story = story_response.json()
            story["username"] = story["by"]
            stories.append(story)
        return json.dumps(stories)

    def run(self, num_stories: int = 5) -> Iterator[RunResponse]:
        # 循環参照を回避するために、ここで hn_agent のツールを設定します
        self.hn_agent.tools = [self.get_top_hackernews_stories]

        logger.info(f"HackerNews からトップ {num_stories} 件のストーリーを取得しています。")
        top_stories: RunResponse = self.hn_agent.run(num_stories=num_stories)
        if top_stories is None or not top_stories.content:
            yield RunResponse(
                run_id=self.run_id, content="申し訳ありませんが、トップストーリーを取得できませんでした。"
            )
            return

        logger.info("各ストーリーを読み取り、レポートを作成しています。")
        yield from self.writer.run(top_stories.content, stream=True)


if __name__ == "__main__":
    # ワークフローを実行します
    report: Iterator[RunResponse] = HackerNewsReporter(
        storage=storage, debug_mode=False
    ).run(num_stories=5)
    # レポートを表示します
    pprint_run_response(report, markdown=True, show_time=True)
```
