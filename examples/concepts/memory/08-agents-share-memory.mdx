---
title: 複数のエージェントによるメモリ共有
---
この例では、同じメモリを共有する2つのエージェントを使用します。

## コード

```python cookbook/agent_concepts/memory/09_agents_share_memory.py

from agno.agent.agent import Agent
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.memory.v2.memory import Memory
from agno.models.google.gemini import Gemini
from agno.tools.duckduckgo import DuckDuckGoTools
from rich.pretty import pprint

memory_db = SqliteMemoryDb(table_name="memory", db_file="tmp/memory.db")

# モデルを設定する必要はありません。エージェントによってエージェントのモデルに設定されます。
memory = Memory(db=memory_db)

# この例のためにメモリをリセットします。
memory.clear()

john_doe_id = "john_doe@example.com"

chat_agent = Agent(
    model=Gemini(id="gemini-2.0-flash-exp"),
    description="ユーザーとチャットできる便利なアシスタントです",
    memory=memory,
    enable_user_memories=True,
)

chat_agent.print_response(
    "私の名前はジョン・ドウです。週末は山でハイキングするのが好きです。",
    stream=True,
    user_id=john_doe_id,
)

chat_agent.print_response("私の趣味は何ですか？", stream=True, user_id=john_doe_id)


research_agent = Agent(
    model=Gemini(id="gemini-2.0-flash-exp"),
    description="ユーザーの研究に関する質問を支援できるリサーチアシスタントです",
    tools=[DuckDuckGoTools(cache_results=True)],
    memory=memory,
    enable_user_memories=True,
)

research_agent.print_response(
    "量子コンピューティングに関する質問をするのが好きです。量子コンピューティングに関する最新ニュースは何ですか？",
    stream=True,
    user_id=john_doe_id,
)

memories = memory.get_user_memories(user_id=john_doe_id)
print("ジョン・ドウに関するメモリ:")
pprint(memories)
```
