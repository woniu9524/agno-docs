title: マルチユーザー・マルチセッションチャット

この例では、マルチユーザー、マルチセッションのチャットを実行する方法を示します。

この例では、3人のユーザーと4つのセッションがあります。

- ユーザー1は2つのセッションを持っています。
- ユーザー2は1つのセッションを持っています。
- ユーザー3は1つのセッションを持っています。

## コード

```python cookbook/agent_concepts/memory/11_multi_user_multi_session_chat.py

import asyncio

from agno.agent.agent import Agent
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.memory.v2.memory import Memory
from agno.models.google.gemini import Gemini
from agno.storage.sqlite import SqliteStorage

agent_storage = SqliteStorage(
    table_name="agent_sessions", db_file="tmp/persistent_memory.db"
)
memory_db = SqliteMemoryDb(table_name="memory", db_file="tmp/memory.db")

memory = Memory(db=memory_db)

# この例のためにメモリをリセットします
memory.clear()

user_1_id = "user_1@example.com"
user_2_id = "user_2@example.com"
user_3_id = "user_3@example.com"

user_1_session_1_id = "user_1_session_1"
user_1_session_2_id = "user_1_session_2"
user_2_session_1_id = "user_2_session_1"
user_3_session_1_id = "user_3_session_1"

chat_agent = Agent(
    model=Gemini(id="gemini-2.0-flash-exp"),
    storage=agent_storage,
    memory=memory,
    enable_user_memories=True,
)


async def run_chat_agent():
    await chat_agent.aprint_response(
        "私の名前はマーク・ゴンザレスです。アニメとビデオゲームが好きです。",
        user_id=user_1_id,
        session_id=user_1_session_1_id,
    )
    await chat_agent.aprint_response(
        "マンガを読むことやビデオゲームをプレイすることも楽しんでいます。",
        user_id=user_1_id,
        session_id=user_1_session_1_id,
    )

    # ユーザー1とのチャット - セッション2
    await chat_agent.aprint_response(
        "今夜映画に行く予定です。",
        user_id=user_1_id,
        session_id=user_1_session_2_id,
    )

    # ユーザー2とのチャット
    await chat_agent.aprint_response(
        "こんにちは、私の名前はジョン・ドウです。", user_id=user_2_id, session_id=user_2_session_1_id
    )
    await chat_agent.aprint_response(
        "今週末はハイキングに行く予定です。",
        user_id=user_2_id,
        session_id=user_2_session_1_id,
    )

    # ユーザー3とのチャット
    await chat_agent.aprint_response(
        "こんにちは、私の名前はジェーン・スミスです。", user_id=user_3_id, session_id=user_3_session_1_id
    )
    await chat_agent.aprint_response(
        "明日ジムに行く予定です。",
        user_id=user_3_id,
        session_id=user_3_session_1_id,
    )

    # ユーザー1との会話を続ける
    # エージェントはユーザー1のすべてのメモリを考慮に入れる必要があります。
    await chat_agent.aprint_response(
        "今週末は何をしたらいいと思いますか？",
        user_id=user_1_id,
        session_id=user_1_session_1_id,
    )


if __name__ == "__main__":
    # ユーザー1とのチャット - セッション1
    asyncio.run(run_chat_agent())

    user_1_memories = memory.get_user_memories(user_id=user_1_id)
    print("ユーザー1のメモリ:")
    for i, m in enumerate(user_1_memories):
        print(f"{i}: {m.memory}")

    user_2_memories = memory.get_user_memories(user_id=user_2_id)
    print("ユーザー2のメモリ:")
    for i, m in enumerate(user_2_memories):
        print(f"{i}: {m.memory}")

    user_3_memories = memory.get_user_memories(user_id=user_3_id)
    print("ユーザー3のメモリ:")
    for i, m in enumerate(user_3_memories):
        print(f"{i}: {m.memory}")
```
