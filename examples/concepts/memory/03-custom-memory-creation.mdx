---
title: カスタムメモリの作成
---

この例では、テキストまたはメッセージのリストを提供することで、エージェントを使用してユーザーメモリを作成する方法を示します。エージェントはカスタムメモリマネージャーを使用して、関連する詳細をキャプチャおよび保存します。

## コード

```python cookbook/agent_concepts/memory/04_custom_memory_creation.py
from agno.memory.v2 import Memory
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.memory.v2.manager import MemoryManager
from agno.models.anthropic.claude import Claude
from agno.models.google import Gemini
from agno.models.message import Message
from rich.pretty import pprint

memory_db = SqliteMemoryDb(table_name="memory", db_file="tmp/memory.db")
# この例のためにリセットします
memory_db.clear()

memory = Memory(
    model=Gemini(id="gemini-2.0-flash-exp"),
    memory_manager=MemoryManager(
        model=Gemini(id="gemini-2.0-flash-exp"),
        memory_capture_instructions="""\
                    ユーザーの学術的興味に関する詳細のみを記憶に含めてください。
                    興味のある科目のみを含めてください。
                    名前、趣味、個人的な興味は無視してください。
                    """,
    ),
    db=memory_db,
)

john_doe_id = "john_doe@example.com"

memory.create_user_memories(
    message="""\
私の名前はジョン・ドウです。

週末は山でハイキングをしたり、
寝る前にSF小説を読んだり、
異文化の新しいレシピを料理したり、
友達とチェスをしたりするのが好きです。

宇宙の歴史やその他の天文学的なトピックについて学ぶことに興味があります。
""",
    user_id=john_doe_id,
)


memories = memory.get_user_memories(user_id=john_doe_id)
print("ジョン・ドウの記憶:")
pprint(memories)


# デフォルトのメモリマネージャーを使用します
memory = Memory(model=Claude(id="claude-3-5-sonnet-latest"), db=memory_db)
jane_doe_id = "jane_doe@example.com"

# メッセージ履歴を送信し、メモリを追加します
memory.create_user_memories(
    messages=[
        Message(role="user", content="こんにちは、お元気ですか？"),
        Message(role="assistant", content="元気です、ありがとう！"),
        Message(role="user", content="何ができますか？"),
        Message(
            role="assistant",
            content="宿題を手伝ったり、宇宙に関する質問に答えたりできます。",
        ),
        Message(role="user", content="私の名前はジェーン・ドウです"),
        Message(role="user", content="チェスをするのが好きです"),
        Message(
            role="user",
            content="実は、チェスをするのが好きだということは忘れてください。ダンジョンズ＆ドラゴンズのようなテーブルトップゲームをプレイする方が好きです。",
        ),
        Message(
            role="user",
            content="宇宙の歴史やその他の天文学的なトピックについて学ぶことにも興味があります。",
        ),
        Message(role="assistant", content="それは素晴らしいですね！"),
        Message(
            role="user",
            content="物理学に本当に興味があります。量子力学について教えてください。",
        ),
    ],
    user_id=jane_doe_id,
)

memories = memory.get_user_memories(user_id=jane_doe_id)
print("ジェーン・ドウの記憶:")
pprint(memories)
```
