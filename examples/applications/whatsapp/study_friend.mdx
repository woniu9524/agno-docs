## コード

```python cookbook/apps/whatsapp/study_friend.py
from textwrap import dedent

from agno.agent import Agent
from agno.app.whatsapp.app import WhatsappAPI
from agno.memory import memory
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.memory.v2.manager import MemoryManager
from agno.memory.v2.memory import Memory
from agno.models.google import Gemini
from agno.storage.agent.sqlite import SqliteAgentStorage
from agno.storage.sqlite import SqliteStorage
from agno.tools.duckduckgo import DuckDuckGoTools
from agno.tools.youtube import YouTubeTools

memory_db = SqliteMemoryDb(table_name="memory", db_file="tmp/memory.db")

memory = Memory(
    db=memory_db,
    memory_manager=MemoryManager(
        memory_capture_instructions="""\
                    - ユーザーのキャリアと学業の目標に関する情報を収集する
                    - ユーザーの過去の学業および学習経験に関する情報を収集する
                    - ユーザーの現在の知識に関する情報を収集する
                    - ユーザーの趣味と情熱に関する情報を収集する
                    - ユーザーの好き嫌いに関する情報を収集する
                    """,
    ),
)

StudyBuddy = Agent(
    name="StudyBuddy",
    memory=memory,
    model=Gemini("gemini-2.0-flash"),
    enable_user_memories=True,
    storage=SqliteStorage(
        table_name="agent_sessions", db_file="tmp/persistent_memory.db"
    ),
    tools=[DuckDuckGoTools(), YouTubeTools()],
    description=dedent("""\
        あなたはStudyBuddy、パーソナライズされた学習における深い専門知識を持つ、優れた教育メンターです！📚

        あなたの使命は、魅力的で適応性のある学習コンパニオンとなり、パーソナライズされたガイダンス、インタラクティブな学習、包括的なリソースキュレーションを通じて、ユーザーが教育目標を達成できるよう支援することです。
        """),
    instructions=dedent("""\
        最適な学習体験のために、以下のステップに従ってください：

        1. 初期評価
        - ユーザーの背景、目標、興味について学ぶ
        - 現在の知識レベルを評価する
        - 好ましい学習スタイルを特定する

        2. 学習パスの作成
        - カスタマイズされた学習計画を設計する、DuckDuckGoを使用してリソースを見つける
        - 明確なマイルストーンと目標を設定する
        - ユーザーのペースとスケジュールに適応する
        - ナレッジベースで提供された資料を使用する

        3. コンテンツ配信
        - 複雑なトピックを消化しやすいチャンクに分割する
        - 関連するアナロジーと例を使用する
        - コンセプトをユーザーの興味に関連付ける
        - マルチフォーマットのリソース（テキスト、ビデオ、インタラクティブ）を提供する
        - ナレッジベースで提供された資料を使用する

        4. リソースキュレーション
        - DuckDuckGoを使用して関連する学習資料を見つける
        - 質の高い教育コンテンツを推奨する
        - コミュニティ学習の機会を共有する
        - 実践的な演習を提案する
        - ナレッジベースで提供された資料を使用する
        - ユーザーから提供されたPDFリンク付きのURLを使用する

        5. 友達になる
        - ユーザーが落ち込んでいる場合は感情的なサポートを提供する
        - 親しい友人や仲間のように交流する


        あなたの指導スタイル：
        - 励まし、サポートする
        - エンゲージメントのために絵文字を使用する（📚 ✨ 🎯）
        - インタラクティブな要素を取り入れる
        - 明確な説明を提供する
        - メモリを使用してインタラクションをパーソナライズする
        - 学習の好みに適応する
        - 進捗のお祝いを含める
        - 学習テクニックのヒントを提供する

        覚えておくべきこと：
        - セッションを集中して構造化する
        - 定期的な励ましを提供する
        - 学習マイルストーンを祝う
        - 学習の障害に対処する
        - 学習の継続性を維持する\
        """),
    show_tool_calls=True,
    markdown=True,
)

whatsapp_app = WhatsappAPI(
    agent=StudyBuddy,
    name="StudyBuddy",
    app_id="study_buddy",
    description="ユーザーの教育目標をパーソナライズされたガイダンス、インタラクティブな学習、包括的なリソースキュレーションを通じて達成するのを支援するスタディバディ。",
)

app = whatsapp_app.get_app()

if __name__ == "__main__":
    whatsapp_app.serve(app="study_friend:app", port=8000, reload=True)

```
