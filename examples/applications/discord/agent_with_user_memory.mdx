## コード

```python cookbook/apps/discord/agent_with_user_memory.py
from textwrap import dedent

from agno.agent import Agent
from agno.app.discord import DiscordClient
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.memory.v2.manager import MemoryManager
from agno.memory.v2.memory import Memory
from agno.models.google import Gemini
from agno.storage.sqlite import SqliteStorage
from agno.tools.googlesearch import GoogleSearchTools

agent_storage = SqliteStorage(
    table_name="agent_sessions", db_file="tmp/persistent_memory.db"
)
memory_db = SqliteMemoryDb(table_name="memory", db_file="tmp/memory.db")

memory = Memory(
    db=memory_db,
    memory_manager=MemoryManager(
        memory_capture_instructions="""\
                        ユーザーの名前を収集する、
                        ユーザーの情熱と趣味に関する情報を収集する、
                        ユーザーの好き嫌いに関する情報を収集する、
                        ユーザーが現在人生で何をしているかについての情報を収集する
                    """,
        model=Gemini(id="gemini-2.0-flash"),
    ),
)


# この例のメモリをリセットします
memory.clear()

personal_agent = Agent(
    name="Basic Agent",
    model=Gemini(id="gemini-2.0-flash"),
    tools=[GoogleSearchTools()],
    add_history_to_messages=True,
    num_history_responses=3,
    add_datetime_to_instructions=True,
    markdown=True,
    memory=memory,
    enable_agentic_memory=True,
    instructions=dedent("""
        あなたはユーザーの個人的なAIの友人です。あなたの目的は、ユーザーと物事についてチャットし、ユーザーが良い気分になるようにすることです。
        まず自己紹介をし、名前を尋ねてから、自分自身、趣味、好きなこと、話したいことについて尋ねてください。
        会話の中の最新情報を検索するためにGoogle検索ツールを使用してください。
                        """),
    debug_mode=True,
)

discord_agent = DiscordClient(personal_agent)
if __name__ == "__main__":
    discord_agent.serve()
```
