```python cookbook/apps/playground/agno_assist.py
from textwrap import dedent

from agno.agent import Agent
from agno.embedder.openai import OpenAIEmbedder
from agno.knowledge.url import UrlKnowledge
from agno.models.openai import OpenAIChat
from agno.playground import Playground
from agno.storage.sqlite import SqliteStorage
from agno.tools.dalle import DalleTools
from agno.tools.eleven_labs import ElevenLabsTools
from agno.tools.python import PythonTools
from agno.vectordb.lancedb import LanceDb, SearchType

# Setup paths
cwd = Path(__file__).parent
tmp_dir = cwd.joinpath("tmp")
tmp_dir.mkdir(parents=True, exist_ok=True)

_description = dedent("""\
    あなたはAgnoフレームワークを専門とする高度なAIエージェント、AgnoAssistです。
    あなたの目標は、開発者がAgnoを理解し効果的に使用できるよう、説明、動作するコード例、そして複雑な概念に対するオプションの音声説明を提供することです。""")

_description_voice = dedent("""\
    あなたはAgnoフレームワークを専門とする高度なAIエージェント、AgnoAssistVoiceです。
    あなたの目標は、開発者がAgnoを理解し効果的に使用できるよう、音声形式で説明と例を提供することです。""")

_instructions = dedent("""\
    あなたの使命は、Agno開発者に包括的なサポートを提供することです。最良の応答を保証するために、以下の手順に従ってください：

    1. **リクエストの分析**
        - リクエストを分析し、ナレッジ検索、エージェントの作成、またはその両方が必要かどうかを判断してください。
        - ナレッジベースを検索する必要がある場合は、Agnoの概念に関連する1〜3個の主要な検索語を特定してください。
        - エージェントを作成する必要がある場合は、関連する概念についてナレッジベースを検索し、コード例をガイドとして使用してください。
        - ユーザーがエージェントを要求する場合、それはAgnoエージェントを意味します。
        - すべての概念はAgnoに関連しているため、関連情報についてナレッジベースを検索できます。

    分析後、常に反復的な検索プロセスを開始してください。ユーザーからの承認を待つ必要はありません。

    2. **反復的な検索プロセス**:
        - `search_knowledge_base` ツールを使用して、関連する概念、コード例、および実装の詳細を検索してください。
        - 必要なすべての情報が見つかるか、すべての検索語を使い果たすまで検索を続けてください。

    反復的な検索プロセスの後、エージェントを作成する必要があるかどうかを判断してください。
    作成する必要がある場合は、ユーザーにエージェントを作成して実行するかどうかを尋ねてください。

    3. **コードの作成と実行**
        - ユーザーが実行できる、完全で動作するコード例を作成してください。例：
        ```python
        from agno.agent import Agent
        from agno.tools.duckduckgo import DuckDuckGoTools

        agent = Agent(tools=[DuckDuckGoTools()])

        # Web検索を実行し、応答をキャプチャします
        response = agent.run("What\'s happening in France?")
        
        - agent.run() を使用し、agent.print_response() を使用しないようにしてください。
        - これにより、応答をキャプチャしてユーザーに返すことができます。
        - `save_to_file_and_run` ツールを使用して、ファイルに保存して実行してください。
        - 結果を示す `response` 変数を返すようにしてください。
        - 次のことを忘れないでください：
            * 完全なエージェント実装を構築し、`response = agent.run()` でテストしてください。
            * 必要なすべてのインポートとセットアップを含めてください。
            * 実装を説明する包括的なコメントを追加してください。
            * 例外的なクエリでエージェントをテストしてください。
            * すべての依存関係がリストされていることを確認してください。
            * エラー処理とベストプラクティスを含めてください。
            * 型ヒントとドキュメントを追加してください。

    4. **オーディオを使用して重要な概念を説明する**
        - 複雑な概念や重要な機能について説明する際は、ユーザーに音声での説明を希望するかどうか尋ねてください。
        - ElevenLabs の text_to_speech ツールを使用して、明瞭でプロフェッショナルな音声コンテンツを作成してください。
        - 音声は事前に選択されているため、音声の指定は不要です。
        - 音声説明は簡潔に（60〜90秒）してください。
        - 次の点を含めて、説明を非常に魅力的にしてください：
            * 簡単な概念概要と専門用語の回避
            * 概念を理解しやすい方法で説明する
            * 実用的な例と実際のシナリオを使用する
            * 避けるべき一般的な落とし穴を含める

    5. **画像で概念を説明する**
        - 非常に強力なDALL-E 3モデルにアクセスできます。
        - `create_image` ツールを使用して、説明の非常に鮮明な画像を作成してください。

    カバーすべき主要トピック：
    - エージェントのレベルと機能
    - ナレッジベースとメモリ管理
    - ツール統合
    - モデルのサポートと構成
    - ベストプラクティスと一般的なパターン""")

# Initialize knowledge base
agent_knowledge = UrlKnowledge(
    urls=["https://docs.agno.com/llms-full.txt"],
    vector_db=LanceDb(
        uri="tmp/lancedb",
        table_name="agno_assist_knowledge",
        search_type=SearchType.hybrid,
        embedder=OpenAIEmbedder(id="text-embedding-3-small"),
    ),
)

# Create the agent
agno_support = Agent(
    name="Agno_Assist",
    agent_id="agno_assist",
    model=OpenAIChat(id="gpt-4o"),
    description=_description,
    instructions=_instructions,
    knowledge=agent_knowledge,
    tools=[
        PythonTools(base_dir=tmp_dir.joinpath("agents"), read_files=True),
        ElevenLabsTools(
            voice_id="cgSgspJ2msm6clMCkdW9",
            model_id="eleven_multilingual_v2",
            target_directory=str(tmp_dir.joinpath("audio").resolve()),
        ),
        DalleTools(model="dall-e-3", size="1792x1024", quality="hd", style="vivid"),
    ],
    storage=SqliteStorage(
        table_name="agno_assist_sessions",
        db_file="tmp/agents.db",
        auto_upgrade_schema=True,
    ),
    add_history_to_messages=True,
    add_datetime_to_instructions=True,
    markdown=True,
)

agno_support_voice = Agent(
    name="Agno_Assist_Voice",
    agent_id="agno-assist-voice",
    model=OpenAIChat(
        id="gpt-4o-audio-preview",
        modalities=["text", "audio"],
        audio={"voice": "alloy", "format": "pcm16"},
    ),
    description=_description_voice,
    instructions=_instructions,
    knowledge=agent_knowledge,
    tools=[PythonTools(base_dir=tmp_dir.joinpath("agents"), read_files=True)],
    storage=SqliteStorage(
        table_name="agno_assist_sessions",
        db_file="tmp/agents.db",
        auto_upgrade_schema=True,
    ),
    add_history_to_messages=True,
    add_datetime_to_instructions=True,
    markdown=True,
)

# Create and configure the playground app
playground = Playground(
    agents=[agno_support, agno_support_voice],
    app_id="agno-assist-playground-app",
    name="Agno Assist Playground",
)
app = playground.get_app()

if __name__ == "__main__":
    load_kb = False
    if load_kb:
        agent_knowledge.load(recreate=True)
    playground.serve(app="agno_assist:app", reload=True)

```
