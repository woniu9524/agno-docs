## Code

```python cookbook/apps/slack/agent_with_user_memory.py
from textwrap import dedent
from agno.agent import Agent
from agno.app.slack.app import SlackAPI
from agno.memory.v2.db.sqlite import SqliteMemoryDb
from agno.memory.v2.manager import MemoryManager
from agno.memory.v2.memory import Memory
from agno.models.anthropic.claude import Claude
from agno.storage.sqlite import SqliteStorage
from agno.tools.googlesearch import GoogleSearchTools

agent_storage = SqliteStorage(
    table_name="agent_sessions", db_file="tmp/persistent_memory.db"
)
memory_db = SqliteMemoryDb(table_name="memory", db_file="tmp/memory.db")

memory = Memory(
    db=memory_db,
    memory_manager=MemoryManager(
        memory_capture_instructions="""\
                        ユーザーの名前を収集する、
                        ユーザーの情熱と趣味に関する情報を収集する、
                        ユーザーの好き嫌いに関する情報を収集する、
                        ユーザーが現在人生で何をしているかに関する情報を収集する
                    """,
        model=Claude(id="claude-3-5-sonnet-20241022"),
    ),
)


# Reset the memory for this example
memory.clear()

personal_agent = Agent(
    name="Basic Agent",
    model=Claude(id="claude-sonnet-4-20250514"),
    tools=[GoogleSearchTools()],
    add_history_to_messages=True,
    num_history_responses=3,
    add_datetime_to_instructions=True,
    markdown=True,
    memory=memory,
    enable_user_memories=True,
    instructions=dedent("""
        あなたはSlackチャットのパーソナルAIフレンドです。目的はユーザーと様々なことについてチャットし、ユーザーに気分良く過ごしてもらうことです。
        まず自己紹介をして、名前を尋ねてください。その後、ユーザー自身のこと、趣味、好きなこと、話したいことについて尋ねてください。
        会話の中の最新情報を調べるためにGoogle検索ツールを使用してください。
        時々、グループメッセージが先頭に付いたメッセージを受信することがあります。その場合は、単一のユーザーからのメッセージとして扱うのではなく、グループ全体に返信してください。
                        """),
    debug_mode=True,
    add_state_in_messages=True,
)


slack_api_app = SlackAPI(
    agent=personal_agent,
    name="Agent with User Memory",
    app_id="agent_with_user_memory",
    description="A agent with user memory that can chat with the user about things and make them feel good.",
)
app = slack_api_app.get_app()

if __name__ == "__main__":
    slack_api_app.serve("agent_with_user_memory:app", port=8000, reload=True)
```
