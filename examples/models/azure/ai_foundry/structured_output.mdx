## コード

```python cookbook/models/azure/ai_foundry/structured_output.py
from typing import List

from agno.agent import Agent, RunResponse  # noqa
from agno.models.azure import AzureAIFoundry
from pydantic import BaseModel, Field
from rich.pretty import pprint  # noqa


class MovieScript(BaseModel):
    setting: str = Field(
        ..., description="ブロックバスター映画の素敵な設定を提供してください。"
    )
    ending: str = Field(
        ...,
        description="映画のエンディング。利用できない場合は、ハッピーエンドを提供してください。",
    )
    genre: str = Field(
        ...,
        description="映画のジャンル。利用できない場合は、アクション、スリラー、またはロマンティックコメディを選択してください。",
    )
    name: str = Field(..., description="この映画に名前を付けてください")
    characters: List[str] = Field(..., description="この映画のキャラクターの名前。")
    storyline: str = Field(
        ..., description="映画の3文のストーリーライン。エキサイティングにしてください！"
    )


agent = Agent(
    model=AzureAIFoundry(id="Phi-4"),
    description="あなたは映画の脚本を書くのを手伝います。",
    response_model=MovieScript,
    # debug_mode=True,
)

# 変数でレスポンスを取得します
# run: RunResponse = agent.run("New York")
# pprint(run.content)

agent.print_response("New York")
```
