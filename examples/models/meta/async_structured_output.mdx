## コード

```python cookbook/models/meta/async_structured_output.py
import asyncio
from typing import List
from pydantic import BaseModel, Field

from agno.agent import Agent
from agno.models.meta import Llama

class MovieScript(BaseModel):
    name: str = Field(..., description="映画の名前。")
    setting: str = Field(..., description="映画の設定を提供してください。")
    ending: str = Field(..., description="映画の結末を説明してください。")
    genre: str = Field(..., description="映画のジャンル。")
    characters: List[str] = Field(..., description="キャラクターのリスト。")
    storyline: str = Field(..., description="3文のストーリーライン。")

agent = Agent(
    model=Llama(id="Llama-3.3-70B"),
    response_model=MovieScript,
    markdown=True,
)

asyncio.run(
    agent.aprint_response(
        "SFアドベンチャーの映画スクリプトのアウトラインを生成してください。"
    )
)
```
