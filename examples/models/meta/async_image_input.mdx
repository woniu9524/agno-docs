import asyncio
from agno.agent import Agent
from agno.media import Image
from agno.models.meta import Llama

async def main():
    agent = Agent(
        model=Llama(id="Llama-4-Scout-17B"),
        markdown=True,
    )

    await agent.aprint_response(
        "请描述这张图片中的场景。",

asyncio.run(main())
