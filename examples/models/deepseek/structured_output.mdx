## 代码

```python cookbook/models/deepseek/structured_output.py
from typing import List

from agno.agent import Agent, RunResponse  # noqa
from agno.models.deepseek import DeepSeek
from pydantic import BaseModel, Field
from rich.pretty import pprint  # noqa


class MovieScript(BaseModel):
    setting: str = Field(
        ..., description="为一部大片提供一个精彩的场景。"
    )
    ending: str = Field(
        ...,
        description="电影的结局。如果没有，请提供一个美好结局。",
    )
    genre: str = Field(
        ...,
        description="电影的类型。如果没有，请选择动作、惊悚或浪漫喜剧。",
    )
    name: str = Field(..., description="给这部电影起个名字")
    characters: List[str] = Field(..., description="这部电影的角色名字。")
    storyline: str = Field(
        ..., description="这部电影三句话的剧情。要写得激动人心！"
    )


json_mode_agent = Agent(
    model=DeepSeek(id="deepseek-chat"),
    description="你帮助人们写电影剧本。",
    response_model=MovieScript,
)

# Get the response in a variable
# json_mode_response: RunResponse = json_mode_agent.run("New York")
# pprint(json_mode_response.content)

json_mode_agent.print_response("New York")
```
