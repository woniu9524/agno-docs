## 代码

```python cookbook/models/xai/structured_output.py
import asyncio
from typing import List

from agno.agent import Agent
from agno.models.xai import xAI
from agno.run.response import RunResponse
from pydantic import BaseModel, Field


class MovieScript(BaseModel):
    name: str = Field(..., description="给这部电影起个名字")
    setting: str = Field(
        ..., description="为一部大片提供一个好的背景设定。"
    )
    ending: str = Field(
        ...,
        description="电影的结局。如果没有，请给一个大团圆结局。",
    )
    genre: str = Field(
        ...,
        description="电影的类型。如果没有，请选择动作、惊悚或浪漫喜剧。",
    )
    characters: List[str] = Field(..., description="这部电影的角色名字。")
    storyline: str = Field(
        ..., description="电影的三句话剧情梗概。要令人兴奋！"
    )


# 返回结构化输出的代理
structured_output_agent = Agent(
    model=xAI(id="grok-2-latest"),
    description="你负责写电影剧本。",
    response_model=MovieScript,
)

structured_output_agent.print_response("羊驼统治世界")
```
