## 代码

```python cookbook/models/google/gemini/structured_output.py
from typing import List

from agno.agent import Agent, RunResponse  # noqa
from agno.models.google import Gemini
from pydantic import BaseModel, Field
from rich.pretty import pprint  # noqa


class MovieScript(BaseModel):
    setting: str = Field(
        ..., description="为一部大片提供一个不错的场景。"
    )
    ending: str = Field(
        ...,
        description="电影的结局。如果不可用，请提供一个快乐的结局。",
    )
    genre: str = Field(
        ...,
        description="电影的类型。如果不可用，请选择动作片、惊悚片或浪漫喜剧。",
    )
    name: str = Field(..., description="给这部电影起个名字")
    characters: List[str] = Field(..., description="这部电影的角色名字。")
    storyline: str = Field(
        ..., description="这部电影的三句话剧情梗概。要精彩！"
    )


movie_agent = Agent(
    model=Gemini(id="gemini-2.0-flash-exp"),
    description="你帮助人们编写电影剧本。",
    response_model=MovieScript,
)

movie_agent.print_response("纽约")
```
