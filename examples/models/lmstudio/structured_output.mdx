import asyncio
from typing import List

from agno.agent import Agent
from agno.models.lmstudio import LMStudio
from pydantic import BaseModel, Field


class MovieScript(BaseModel):
    name: str = Field(..., description="给这部电影起个名字")
    setting: str = Field(
        ..., description="为一部大片提供一个不错的场景。"
    )
    ending: str = Field(
        ...,
        description="电影的结局。如果不可用，请提供一个快乐的结局。",
    )
    genre: str = Field(
        ...,
        description="电影的类型。如果不可用，请选择动作、惊悚或浪漫喜剧。",
    )
    characters: List[str] = Field(..., description="这部电影的角色名字。")
    storyline: str = Field(
        ..., description="电影的3句话剧情梗概。要激动人心！"
    )


# 返回结构化输出的 Agent
structured_output_agent = Agent(
    model=LMStudio(id="qwen2.5-7b-instruct-1m"),
    description="你负责编写电影剧本。",
    response_model=MovieScript,
)
