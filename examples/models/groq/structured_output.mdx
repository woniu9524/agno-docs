## Code

```python cookbook/models/groq/structured_output.py
from typing import List

from agno.agent import Agent, RunResponse  # noqa
from agno.models.groq import Groq
from pydantic import BaseModel, Field
from rich.pretty import pprint  # noqa


class MovieScript(BaseModel):
    setting: str = Field(
        ..., description="为一部大片提供一个漂亮的场景。"
    )
    ending: str = Field(
        ...,
        description="电影的结局。如果不确定，请提供一个快乐的结局。",
    )
    genre: str = Field(
        ...,
        description="电影的类型。如果不确定，请选择动作、惊悚或浪漫喜剧。",
    )
    name: str = Field(..., description="为这部电影起一个名字")
    characters: List[str] = Field(..., description="这部电影的角色名字列表。")
    storyline: str = Field(
        ..., description="关于这部电影的 3 句话故事梗概。要写得激动人心！"
    )


json_mode_agent = Agent(
    model=Groq(id="llama-3.3-70b-versatile"),
    description="你帮助人们编写电影剧本。",
    response_model=MovieScript,
)

json_mode_agent.print_response("New York")
```
