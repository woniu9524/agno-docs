## Code

```python cookbook/models/litellm/storage.py
from agno.agent import Agent
from agno.models.litellm import LiteLLM
from agno.storage.sqlite import SqliteStorage
from agno.tools.duckduckgo import DuckDuckGoTools

# SQLiteデータベースを使用してストレージバックエンドを作成します
storage = SqliteStorage(
    # セッションをai.sessionsテーブルに保存します
    table_name="agent_sessions_storage",
    # db_file: SQLiteデータベースファイル
    db_file="tmp/data.db",
)

# Agentにストレージを追加します
agent = Agent(
    model=LiteLLM(id="gpt-4o"),
    storage=storage,
    tools=[DuckDuckGoTools()],
    add_history_to_messages=True,
)

agent.print_response("カナダには何人住んでいますか？")
agent.print_response("国歌の名前は何ですか？")
```
