## コード

```python cookbook/models/together/structured_output.py
from typing import List

from agno.agent import Agent, RunResponse  # noqa
from agno.models.together import Together
from pydantic import BaseModel, Field
from rich.pretty import pprint  # noqa


class MovieScript(BaseModel):
    setting: str = Field(
        ..., description="ブロックバスター映画の素敵な設定を提供してください。"
    )
    ending: str = Field(
        ...,
        description="映画のエンディング。利用できない場合は、ハッピーエンドを提供してください。",
    )
    genre: str = Field(
        ...,
        description="映画のジャンル。利用できない場合は、アクション、スリラー、またはロマンティックコメディを選択してください。",
    )
    name: str = Field(..., description="この映画に名前を付けてください")
    characters: List[str] = Field(..., description="この映画のキャラクターの名前。")
    storyline: str = Field(
        ..., description="映画の3行のストーリーライン。エキサイティングにしてください！"
    )


# JSONモードを使用するエージェント
json_mode_agent = Agent(
    model=Together(id="meta-llama/Meta-Llama-3.1-8B-Instruct-Turbo"),
    description="映画の脚本を書きます。",
    response_model=MovieScript,
)

# レスポンスを変数に格納
# json_mode_response: RunResponse = json_mode_agent.run("New York")
# pprint(json_mode_response.content)

json_mode_agent.print_response("New York")
```
