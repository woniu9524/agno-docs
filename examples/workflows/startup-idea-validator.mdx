title: Startup Idea Validator
---

このワークフローは、起業家が以下のステップを通じてスタートアップのアイデアを検証するのに役立ちます。
1. 主要なビジネスコンセプトの明確化と洗練
2. 既存のソリューションとの比較による独自性の評価
3. 明確なミッションと目標の定義
4. 包括的な市場調査と分析の実施

なぜこれが役立つのか？
--------------------------------------------------------------------------------
• リソースを投資する前に、スタートアップのアイデアについて客観的なフィードバックを得る
• 総獲得可能市場（TAM）とターゲットセグメントを理解する
• 市場機会と競合に関する仮説を検証する
• 実行を導くための明確なミッションと目標を定義する

誰が使うべきか？
--------------------------------------------------------------------------------
• 起業家およびスタートアップ創業者
• プロダクトマネージャーおよびビジネスストラテジスト
• イノベーションチーム
• エンジェル投資家およびVC（初期スクリーニング実施者）

使用例：
--------------------------------------------------------------------------------
• 新製品/サービスの検証
• 市場機会評価
• 競合分析
• ビジネスモデル検証
• ターゲット顧客セグメンテーション
• ミッション/ビジョン洗練

クイックスタート：
--------------------------------------------------------------------------------
1. 依存関係のインストール：
   pip install openai agno

2. 環境変数の設定：
   - export OPENAI_API_KEY="xxx"

3. 実行：
   python startup_idea_validator.py

ワークフローは、AI駆動の分析とリサーチを通じて、スタートアップのアイデアを検証するようにガイドします。このインサイトを使用して、コンセプトとビジネスプランを洗練させてください！
"""

```python startup_idea_validator.py
import json
from typing import Iterator, Optional

from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.storage.sqlite import SqliteStorage
from agno.tools.googlesearch import GoogleSearchTools
from agno.utils.log import logger
from agno.utils.pprint import pprint_run_response
from agno.workflow import RunEvent, RunResponse, Workflow
from pydantic import BaseModel, Field


class IdeaClarification(BaseModel):
    originality: str = Field(..., description="アイデアの独自性。")
    mission: str = Field(..., description="会社のミッション。")
    objectives: str = Field(..., description="会社の目標。")


class MarketResearch(BaseModel):
    total_addressable_market: str = Field(
        ..., description="総獲得可能市場（TAM）。"
    )
    serviceable_available_market: str = Field(
        ..., description="サービス提供可能市場（SAM）。"
    )
    serviceable_obtainable_market: str = Field(
        ..., description="サービス提供可能獲得市場（SOM）。"
    )
    target_customer_segments: str = Field(..., description="ターゲット顧客セグメント。")


class StartupIdeaValidator(Workflow):
    idea_clarifier_agent: Agent = Agent(
        model=OpenAIChat(id="gpt-4o-mini"),
        instructions=[
            "ユーザーのスタートアップのアイデアが与えられたら、そのアイデアを洗練させることがあなたの目標です。",
            "既存のコンセプトと比較して、アイデアの独自性を評価します。",
            "スタートアップのミッションと目標を定義します。",
        ],
        add_history_to_messages=True,
        add_datetime_to_instructions=True,
        response_model=IdeaClarification,
        debug_mode=False,
    )

    market_research_agent: Agent = Agent(
        model=OpenAIChat(id="gpt-4o-mini"),
        tools=[GoogleSearchTools()],
        instructions=[
            "スタートアップのアイデアと会社のミッションおよび目標が提供されます。",
            "総獲得可能市場（TAM）、サービス提供可能市場（SAM）、サービス提供可能獲得市場（SOM）を推定します。",
            "ターゲット顧客セグメントとその特性を定義します。",
            "必要であれば、Webでリソースを検索してください。",
        ],
        add_history_to_messages=True,
        add_datetime_to_instructions=True,
        response_model=MarketResearch,
        debug_mode=False,
    )

    competitor_analysis_agent: Agent = Agent(
        model=OpenAIChat(id="gpt-4o-mini"),
        tools=[GoogleSearchTools()],
        instructions=[
            "スタートアップのアイデアと、そのアイデアに関連する市場調査が提供されます。",
            "市場における既存の競合他社を特定します。",
            "各競合他社について、強み、弱み、機会、脅威（SWOT）分析を実行します。",
            "競合他社と比較した場合のスタートアップの潜在的なポジショニングを評価します。",
        ],
        add_history_to_messages=True,
        add_datetime_to_instructions=True,
        markdown=True,
        debug_mode=False,
    )

    report_agent: Agent = Agent(
        model=OpenAIChat(id="gpt-4o-mini"),
        instructions=[
            "スタートアップのアイデアと、そのアイデアに関するその他のデータが提供されます。",
            "すべてを単一のレポートに要約します。",
        ],
        add_history_to_messages=True,
        add_datetime_to_instructions=True,
        markdown=True,
        debug_mode=False,
    )

    def get_idea_clarification(self, startup_idea: str) -> Optional[IdeaClarification]:
        try:
            response: RunResponse = self.idea_clarifier_agent.run(startup_idea)

            # 有効な応答が得られたか確認
            if not response or not response.content:
                logger.warning("空のアイデア明確化応答")
            # 応答が期待される型であるか確認
            if not isinstance(response.content, IdeaClarification):
                logger.warning("無効な応答タイプ")

            return response.content

        except Exception as e:
            logger.warning(f"失敗しました: {str(e)}")

        return None

    def get_market_research(
        self, startup_idea: str, idea_clarification: IdeaClarification
    ) -> Optional[MarketResearch]:
        agent_input = {"startup_idea": startup_idea, **idea_clarification.model_dump()}

        try:
            response: RunResponse = self.market_research_agent.run(
                json.dumps(agent_input, indent=4)
            )

            # 有効な応答が得られたか確認
            if not response or not response.content:
                logger.warning("空の市場調査応答")

            # 応答が期待される型であるか確認
            if not isinstance(response.content, MarketResearch):
                logger.warning("無効な応答タイプ")

            return response.content

        except Exception as e:
            logger.warning(f"失敗しました: {str(e)}")

        return None

    def get_competitor_analysis(
        self, startup_idea: str, market_research: MarketResearch
    ) -> Optional[str]:
        agent_input = {"startup_idea": startup_idea, **market_research.model_dump()}

        try:
            response: RunResponse = self.competitor_analysis_agent.run(
                json.dumps(agent_input, indent=4)
            )

            # 有効な応答が得られたか確認
            if not response or not response.content:
                logger.warning("空の競合分析応答")

            return response.content

        except Exception as e:
            logger.warning(f"失敗しました: {str(e)}")

        return None

    def run(self, startup_idea: str) -> Iterator[RunResponse]:
        logger.info(f"スタートアップ検証レポートを作成中: {startup_idea}")

        # アイデアの明確化と定量化
        idea_clarification: Optional[IdeaClarification] = self.get_idea_clarification(
            startup_idea
        )

        if idea_clarification is None:
            yield RunResponse(
                event=RunEvent.workflow_completed,
                content=f"申し訳ありませんが、アイデアを明確にできませんでした: {startup_idea}",
            )
            return

        # 市場調査の実施
        market_research: Optional[MarketResearch] = self.get_market_research(
            startup_idea, idea_clarification
        )

        if market_research is None:
            yield RunResponse(
                event=RunEvent.workflow_completed,
                content="市場調査に失敗しました",
            )
            return

        competitor_analysis: Optional[str] = self.get_competitor_analysis(
            startup_idea, market_research
        )

        # 最終レポートのコンパイル
        final_response: RunResponse = self.report_agent.run(
            json.dumps(
                {
                    "startup_idea": startup_idea,
                    **idea_clarification.model_dump(),
                    **market_research.model_dump(),
                    "competitor_analysis_report": competitor_analysis,
                },
                indent=4,
            )
        )

        yield RunResponse(
            content=final_response.content, event=RunEvent.workflow_completed
        )


# スクリプトが直接実行された場合にワークフローを実行
if __name__ == "__main__":
    from rich.prompt import Prompt

    # ユーザーからアイデアを取得
    idea = Prompt.ask(
        "[bold]あなたのスタートアップのアイデアは何ですか？[/bold]\n✨",
        default="革製のクリスマスオーナメントのマーケットプレイス",
    )

    # セッションIDで使用するために、アイデアをURLセーフな文字列に変換
    url_safe_idea = idea.lower().replace(" ", "-")

    startup_idea_validator = StartupIdeaValidator(
        description="スタートアップアイデアバリデーター",
        session_id=f"validate-startup-idea-{url_safe_idea}",
        storage=SqliteStorage(
            table_name="validate_startup_ideas_workflow",
            db_file="tmp/agno_workflows.db",
        ),
    )

    final_report: Iterator[RunResponse] = startup_idea_validator.run(startup_idea=idea)

    pprint_run_response(final_report, markdown=True)
```
