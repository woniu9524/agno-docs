この例では、共有される `team_session_state` がネストされたエージェントやサブチーム全体にどのように伝播・永続化され、共同タスクの状態管理をシームレスに実現できるかを示します。

## コード

```python cookbook/teams/team_with_nested_shared_state.py

from agno.agent.agent import Agent
from agno.models.openai.chat import OpenAIChat
from agno.team import Team


# ショッピングリストを管理するためのツールを定義します
def add_item(agent: Agent, item: str) -> str:
    """ショッピングリストにアイテムを追加し、確認メッセージを返します。

    Args:
        item (str): ショッピングリストに追加するアイテム。
    """
    # アイテムがリストにまだ存在しない場合に追加します
    if item.lower() not in [
        i.lower() for i in agent.team_session_state["shopping_list"]
    ]:
        agent.team_session_state["shopping_list"].append(item)
        return f"'{item}' をショッピングリストに追加しました"
    else:
        return f"'{item}' は既にショッピングリストにあります"


def remove_item(agent: Agent, item: str) -> str:
    """アイテム名を指定してショッピングリストからアイテムを削除します。

    Args:
        item (str): ショッピングリストから削除するアイテム。
    """
    # 大文字小文字を区別しない検索
    for i, list_item in enumerate(agent.team_session_state["shopping_list"]):
        if list_item.lower() == item.lower():
            agent.team_session_state["shopping_list"].pop(i)
            return f"'{list_item}' をショッピングリストから削除しました"

    return f"'{item}' はショッピングリストに見つかりませんでした。現在のショッピングリスト: {agent.team_session_state['shopping_list']}"


def remove_all_items(agent: Agent) -> str:
    """ショッピングリストからすべてのアイテムを削除します。"""
    agent.team_session_state["shopping_list"] = []
    return "ショッピングリストからすべてのアイテムを削除しました"


shopping_list_agent = Agent(
    name="ショッピングリストエージェント",
    role="ショッピングリストを管理する",
    agent_id="shopping_list_manager",
    model=OpenAIChat(id="gpt-4o-mini"),
    tools=[add_item, remove_item, remove_all_items],
    instructions=[
        "ショッピングリストのアイテムの追加と削除を管理する",
        "アイテムが追加または削除されたときは常に確認する",
        "タスクが完了したら、実行した変更とタスクを記録するためにセッション状態を更新する",
    ],
)


# ショッピング管理チーム - ショッピングリストのすべての変更を処理するための新しいレイヤー
shopping_mgmt_team = Team(
    name="ショッピング管理チーム",
    team_id="shopping_management",
    mode="coordinate",
    model=OpenAIChat(id="gpt-4o-mini"),
    show_tool_calls=True,
    members=[shopping_list_agent],
    instructions=[
        "ショッピングリストエージェントを使用してショッピングリストへのアイテムの追加と削除を管理する",
        "アイテムの追加または削除のリクエストをショッピングリストエージェントに転送する",
    ],
)


def get_ingredients(agent: Agent) -> str:
    """レシピの提案に使用するために、ショッピングリストから材料を取得します。

    Args:
        meal_type (str): 提案する食事の種類（朝食、昼食、夕食、軽食、または任意）
    """
    shopping_list = agent.team_session_state["shopping_list"]

    if not shopping_list:
        return "ショッピングリストは空です。レシピの提案を得るには、まず材料を追加してください。"

    # 材料を返すだけです - エージェントがレシピを作成します
    return f"ショッピングリストからの利用可能な材料: {', '.join(shopping_list)}"


recipe_agent = Agent(
    name="レシピ提案エージェント",
    agent_id="recipe_suggester",
    role="利用可能な材料に基づいてレシピを提案する",
    model=OpenAIChat(id="gpt-4o-mini"),
    tools=[get_ingredients],
    instructions=[
        "まず、get_ingredientsツールを使用して、ショッピングリストから現在の材料を取得する",
        "材料を取得した後、それらの材料に基づいて詳細なレシピの提案を作成する",
        "利用可能な材料を使用して、少なくとも3つの異なるレシピのアイデアを作成する",
        "各レシピには以下を含める：名前、必要な材料（ショッピングリストからのものを強調表示）、簡単な準備手順",
        "レシピの提案は創造的かつ実用的であること",
        "ショッピングリストのアイテムに加えて、通常人が利用できる一般的なパントリーアイテムを考慮する",
        "ユーザーが言及した場合、食事の好みも考慮する",
        "食事の種類が指定されていない場合は、さまざまなオプション（朝食、昼食、夕食、軽食）を提案する",
    ],
)


def list_items(team: Team) -> str:
    """ショッピングリストのすべてのアイテムを一覧表示します。"""
    shopping_list = team.team_session_state["shopping_list"]

    if not shopping_list:
        return "ショッピングリストは空です。"

    items_text = "\n".join([f"- {item}" for item in shopping_list])
    return f"現在のショッピングリスト:\n{items_text}"


# 食事計画サブチームを作成します
meal_planning_team = Team(
    name="食事計画チーム",
    team_id="meal_planning",
    mode="coordinate",
    model=OpenAIChat(id="gpt-4o-mini"),
    members=[recipe_agent],
    instructions=[
        "あなたはショッピングリストのアイテムに基づいてレシピを提案する食事計画チームです。",
        "重要：ユーザーが「これらの材料で何が作れますか？」またはレシピ関連の質問をした場合は、追加情報なしで、正確に同じリクエストをレシピエージェントに即座に転送してください。",
        "ユーザーに材料を尋ねないでください - レシピエージェントはショッピングリストに既に存在するもので作業します。",
        "あなたの主な仕事は、レシピのリクエストを直接、変更せずにレシピエージェントに転送することです。",
    ],
)


def add_chore(team: Team, chore: str, priority: str = "medium") -> str:
    """優先度レベル付きでタスクリストにタスクを追加します。

    Args:
        chore (str): リストに追加するタスク
        priority (str): タスクの優先度レベル（低、中、高）

    Returns:
        str: 確認メッセージ
    """
    # 存在しない場合は、choresリストを初期化します
    if "chores" not in team.session_state:
        team.session_state["chores"] = []

    # 優先度を検証します
    valid_priorities = ["low", "medium", "high"]
    if priority.lower() not in valid_priorities:
        priority = "medium"  # 無効な場合はデフォルトでmediumになります

    # タイムスタンプと優先度でタスクを追加します
    from datetime import datetime

    chore_entry = {
        "description": chore,
        "priority": priority.lower(),
        "added_at": datetime.now().strftime("%Y-%m-%d %H:%M"),
    }

    team.session_state["chores"].append(chore_entry)

    return f"タスク '{chore}' を {priority} の優先度で追加しました"


shopping_team = Team(
    name="ショッピングリストチーム",
    mode="coordinate",
    model=OpenAIChat(id="gpt-4o-mini"),
    team_session_state={"shopping_list": []},
    tools=[list_items, add_chore],
    session_state={"chores": []},
    team_id="shopping_list_team",
    members=[
        shopping_mgmt_team,
        meal_planning_team,
    ],
    show_tool_calls=True,
    markdown=True,
    instructions=[
        "あなたはショッピングリストを管理し、そのリストを使用して食事計画を支援するチームです。",
        "ショッピングリストにアイテムを追加または削除する必要がある場合は、完全なリクエストをショッピング管理チームに転送してください。",
        "重要：ユーザーがレシピや材料で何が作れるかについて尋ねた場合は、追加の質問なしで、正確なリクエストを即座に食事計画チームに転送してください。",
        "例：ユーザーが「これらの材料で何が作れますか？」と尋ねた場合、追加情報なしでこの正確なリクエストを食事計画チームに転送するだけでよい。",
        "ショッピングリストのアイテムを一覧表示する必要がある場合は、list_itemsツールを使用してください。",
        "ユーザーがショッピングリストから何かを取得した場合、それはショッピングリストから削除できることを意味します。",
        "各タスク完了後、add_choreツールを使用して、何が行われたかを正確に高優先度で記録してください。",
    ],
    show_members_responses=True,
)

# 使用例
shopping_team.print_response(
    "ミルク、卵、パンをショッピングリストに追加してください", stream=True
)
print(f"セッション状態: {shopping_team.team_session_state}")

shopping_team.print_response("パンをもらいました", stream=True)
print(f"セッション状態: {shopping_team.team_session_state}")

shopping_team.print_response("リンゴとオレンジが必要です", stream=True)
print(f"セッション状態: {shopping_team.team_session_state}")

shopping_team.print_response("私のリストには何がありますか？", stream=True)
print(f"セッション状態: {shopping_team.team_session_state}")

# 食事計画機能を試す
shopping_team.print_response("これらの材料で何が作れますか？", stream=True)
print(f"セッション状態: {shopping_team.team_session_state}")

shopping_team.print_response(
    "リストからすべてをクリアして、バナナとヨーグルトだけでやり直してください",
    stream=True,
)
print(f"共有セッション状態: {shopping_team.team_session_state}")


print(f"チームセッション状態: {shopping_team.session_state}")

```
